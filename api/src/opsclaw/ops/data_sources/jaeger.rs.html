1//! Jaeger trace API data source.
2//!
3//! Queries `/api/services` to list services, then
4//! `/api/traces?service={name}&limit=20` for recent traces. Flags traces
5//! with `error=true` tags or high latency (>2 s).
6
7use anyhow::{Context, Result};
8use serde::{Deserialize, Serialize};
9
10use crate::ops::data_sources::JaegerConfig;
11
12/// High-latency threshold in microseconds (2 seconds).
13const HIGH_LATENCY_US: u64 = 2_000_000;
14
15// ---------------------------------------------------------------------------
16// Public types
17// ---------------------------------------------------------------------------
18
19#[derive(Debug, Clone, Serialize, Deserialize)]
20pub struct TraceSummary {
21    pub trace_id: String,
22    pub service: String,
23    pub span_count: usize,
24    pub duration_ms: u64,
25    pub has_error: bool,
26    pub high_latency: bool,
27}
28
29// ---------------------------------------------------------------------------
30// Jaeger API response shapes
31// ---------------------------------------------------------------------------
32
33#[derive(Debug, Deserialize)]
34struct ServicesResponse {
35    data: Vec<String>,
36}
37
38#[derive(Debug, Deserialize)]
39struct TracesResponse {
40    data: Vec<TraceData>,
41}
42
43#[derive(Debug, Deserialize)]
44struct TraceData {
45    #[serde(rename = "traceID")]
46    trace_id: String,
47    spans: Vec<SpanData>,
48}
49
50#[derive(Debug, Deserialize)]
51struct SpanData {
52    #[serde(default)]
53    duration: u64,
54    #[serde(default)]
55    tags: Vec<Tag>,
56}
57
58#[derive(Debug, Deserialize)]
59struct Tag {
60    key: String,
61    value: serde_json::Value,
62}
63
64// ---------------------------------------------------------------------------
65// Public API
66// ---------------------------------------------------------------------------
67
68/// Fetch all services, then retrieve recent traces and flag problems.
69pub async fn fetch_problem_traces(cfg: &JaegerConfig) -> Result<Vec<TraceSummary>> {
70    let base = cfg.url.trim_end_matches('/');
71    let client = reqwest::Client::new();
72
73    // 1. List services
74    let services: ServicesResponse = client
75        .get(format!("{base}/api/services"))
76        .send()
77        .await
78        .context("failed to reach Jaeger services API")?
79        .error_for_status()
80        .context("Jaeger services API error")?
81        .json()
82        .await
83        .context("failed to parse Jaeger services response")?;
84
85    let mut summaries = Vec::new();
86
87    // 2. For each service, fetch recent traces
88    for svc in &services.data {
89        let traces: TracesResponse = match client
90            .get(format!("{base}/api/traces?service={svc}&limit=20"))
91            .send()
92            .await
93        {
94            Ok(resp) => match resp.error_for_status() {
95                Ok(r) => match r.json().await {
96                    Ok(t) => t,
97                    Err(e) => {
98                        tracing::warn!("jaeger: failed to parse traces for {svc}: {e:#}");
99                        continue;
100                    }
101                },
102                Err(e) => {
103                    tracing::warn!("jaeger: error fetching traces for {svc}: {e:#}");
104                    continue;
105                }
106            },
107            Err(e) => {
108                tracing::warn!("jaeger: could not reach traces API for {svc}: {e:#}");
109                continue;
110            }
111        };
112
113        for trace in &traces.data {
114            let span_count = trace.spans.len();
115            let duration_us = trace.spans.iter().map(|s| s.duration).max().unwrap_or(0);
116            let has_error = trace.spans.iter().any(|s| {
117                s.tags.iter().any(|t| {
118                    t.key == "error"
119                        && match &t.value {
120                            serde_json::Value::Bool(true) => true,
121                            serde_json::Value::String(s) => s == "true",
122                            _ => false,
123                        }
124                })
125            });
126            let high_latency = duration_us > HIGH_LATENCY_US;
127
128            if has_error || high_latency {
129                summaries.push(TraceSummary {
130                    trace_id: trace.trace_id.clone(),
131                    service: svc.clone(),
132                    span_count,
133                    duration_ms: duration_us / 1000,
134                    has_error,
135                    high_latency,
136                });
137            }
138        }
139    }
140
141    Ok(summaries)
142}

148#[cfg(test)]
149mod tests {
150    use super::*;
151
152    #[test]
153    fn deserialize_services_response() {
154        let json = r#"{"data":["frontend","backend"]}"#;
155        let resp: ServicesResponse = serde_json::from_str(json).unwrap();
156        assert_eq!(resp.data, vec!["frontend", "backend"]);
157    }
158
159    #[test]
160    fn deserialize_traces_response() {
161        let json = r#"{"data":[{"traceID":"abc123","spans":[{"duration":1500000,"tags":[{"key":"error","value":true}]}]}]}"#;
162        let resp: TracesResponse = serde_json::from_str(json).unwrap();
163        assert_eq!(resp.data.len(), 1);
164        assert_eq!(resp.data[0].trace_id, "abc123");
165        assert_eq!(resp.data[0].spans[0].duration, 1_500_000);
166    }
167
168    #[test]
169    fn error_tag_detected() {
170        let tag = Tag {
171            key: "error".into(),
172            value: serde_json::Value::Bool(true),
173        };
174        assert!(tag.key == "error" && matches!(&tag.value, serde_json::Value::Bool(true)));
175    }
176
177    #[test]
178    fn high_latency_threshold() {
179        const _: () = assert!(3_000_000u64 > HIGH_LATENCY_US);
180        const _: () = assert!(1_000_000u64 <= HIGH_LATENCY_US);
181    }
182}
