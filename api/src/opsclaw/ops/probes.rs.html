16/// Outcome of a single probe execution.
17#[derive(Debug, Clone)]
18pub struct ProbeResult {
19    pub probe_name: String,
20    pub probe_type: String,
21    pub success: bool,
22    pub latency_ms: u64,
23    pub message: String,
24    pub details: Option<String>,
25}

31/// Run a single probe via the given [`CommandRunner`].
32pub async fn run_probe(runner: &dyn CommandRunner, probe: &ProbeConfig) -> Result<ProbeResult> {
33    match &probe.probe_type {
34        ProbeType::Http {
35            url,
36            expected_status,
37            timeout_secs,
38        } => run_http_probe(runner, &probe.name, url, *expected_status, *timeout_secs).await,
39        ProbeType::Tcp {
40            host,
41            port,
42            timeout_secs,
43        } => run_tcp_probe(runner, &probe.name, host, *port, *timeout_secs).await,
44        ProbeType::Dns {
45            hostname,
46            expected_ip,
47        } => run_dns_probe(runner, &probe.name, hostname, expected_ip.as_deref()).await,
48        ProbeType::TlsCert {
49            hostname,
50            port,
51            warn_days,
52        } => run_tls_probe(runner, &probe.name, hostname, *port, *warn_days).await,
53    }
54}

56async fn run_http_probe(
57    runner: &dyn CommandRunner,
58    name: &str,
59    url: &str,
60    expected_status: Option<u16>,
61    timeout_secs: u64,
62) -> Result<ProbeResult> {
63    let cmd = format!(
64        "curl -sSL -o /dev/null -w '%{{http_code}} %{{time_total}}' --max-time {timeout_secs} '{url}'"
65    );
66    let output = runner.run(&cmd).await?;
67    let (success, latency_ms, message, details) =
68        parse_curl_output(&output.stdout, output.exit_code, expected_status);
69    Ok(ProbeResult {
70        probe_name: name.to_string(),
71        probe_type: "http".to_string(),
72        success,
73        latency_ms,
74        message,
75        details: Some(details),
76    })
77}

79async fn run_tcp_probe(
80    runner: &dyn CommandRunner,
81    name: &str,
82    host: &str,
83    port: u16,
84    timeout_secs: u64,
85) -> Result<ProbeResult> {
86    let cmd = format!("nc -z -w{timeout_secs} {host} {port} 2>&1 && echo OK || echo FAIL");
87    let output = runner.run(&cmd).await?;
88    let stdout = output.stdout.trim();
89    let success = stdout.ends_with("OK");
90    Ok(ProbeResult {
91        probe_name: name.to_string(),
92        probe_type: "tcp".to_string(),
93        success,
94        latency_ms: 0,
95        message: if success {
96            format!("TCP connection to {host}:{port} succeeded")
97        } else {
98            format!("TCP connection to {host}:{port} failed")
99        },
100        details: Some(stdout.to_string()),
101    })
102}

104async fn run_dns_probe(
105    runner: &dyn CommandRunner,
106    name: &str,
107    hostname: &str,
108    expected_ip: Option<&str>,
109) -> Result<ProbeResult> {
110    let cmd = format!("dig +short {hostname}");
111    let output = runner.run(&cmd).await?;
112    let (success, message, details) = parse_dig_output(&output.stdout, expected_ip);
113    Ok(ProbeResult {
114        probe_name: name.to_string(),
115        probe_type: "dns".to_string(),
116        success,
117        latency_ms: 0,
118        message,
119        details: Some(details),
120    })
121}

123async fn run_tls_probe(
124    runner: &dyn CommandRunner,
125    name: &str,
126    hostname: &str,
127    port: u16,
128    warn_days: u32,
129) -> Result<ProbeResult> {
130    let check_secs = u64::from(warn_days) * 86400;
131    let is_bare_ip = hostname.parse::<std::net::IpAddr>().is_ok();
132    if is_bare_ip {
133        tracing::warn!("TLS hostname check skipped for bare IP");
134    }
135    let sni_flag = if is_bare_ip {
136        String::new()
137    } else {
138        format!("-servername {hostname} ")
139    };
140    let cmd = format!(
141        "echo | openssl s_client {sni_flag}-connect {hostname}:{port} 2>/dev/null \
142         | openssl x509 -noout -enddate -checkend {check_secs}"
143    );
144    let output = runner.run(&cmd).await?;
145    let (success, message, details) =
146        parse_openssl_output(&output.stdout, output.exit_code, hostname, warn_days);
147    Ok(ProbeResult {
148        probe_name: name.to_string(),
149        probe_type: "tls_cert".to_string(),
150        success,
151        latency_ms: 0,
152        message,
153        details: Some(details),
154    })
155}

161/// Parse curl `-w '%{http_code} %{time_total}'` output.
162///
163/// Returns `(success, latency_ms, message, details)`.
164pub fn parse_curl_output(
165    stdout: &str,
166    exit_code: i32,
167    expected_status: Option<u16>,
168) -> (bool, u64, String, String) {
169    let trimmed = stdout.trim();
170    let parts: Vec<&str> = trimmed.split_whitespace().collect();
171
172    if exit_code != 0 || parts.len() < 2 {
173        return (
174            false,
175            0,
176            format!("HTTP probe failed (exit code {exit_code})"),
177            trimmed.to_string(),
178        );
179    }
180
181    let status_code: u16 = parts[0].parse().unwrap_or(0);
182    let time_total: f64 = parts[1].parse().unwrap_or(0.0);
183    #[allow(clippy::cast_possible_truncation, clippy::cast_sign_loss)]
184    let latency_ms = (time_total * 1000.0).max(0.0) as u64;
185
186    let expected = expected_status.unwrap_or(200);
187    let success = status_code == expected;
188
189    let message = if success {
190        format!("HTTP {status_code} in {latency_ms}ms")
191    } else {
192        format!("HTTP {status_code} (expected {expected}) in {latency_ms}ms")
193    };
194
195    (success, latency_ms, message, trimmed.to_string())
196}

198/// Parse `dig +short` output. Returns `(success, message, details)`.
199pub fn parse_dig_output(stdout: &str, expected_ip: Option<&str>) -> (bool, String, String) {
200    let lines: Vec<&str> = stdout
201        .lines()
202        .map(str::trim)
203        .filter(|l| !l.is_empty())
204        .collect();
205
206    if lines.is_empty() {
207        return (
208            false,
209            "DNS resolution returned no results".to_string(),
210            stdout.to_string(),
211        );
212    }
213
214    match expected_ip {
215        Some(ip) => {
216            let found = lines.contains(&ip);
217            if found {
218                (
219                    true,
220                    format!("DNS resolved to expected IP {ip}"),
221                    lines.join(", "),
222                )
223            } else {
224                (
225                    false,
226                    format!("DNS resolved to [{}], expected {ip}", lines.join(", ")),
227                    lines.join(", "),
228                )
229            }
230        }
231        None => (
232            true,
233            format!("DNS resolved to {}", lines.join(", ")),
234            lines.join(", "),
235        ),
236    }
237}

239/// Parse openssl cert check output. Returns `(success, message, details)`.
240pub fn parse_openssl_output(
241    stdout: &str,
242    exit_code: i32,
243    hostname: &str,
244    warn_days: u32,
245) -> (bool, String, String) {
246    let trimmed = stdout.trim();
247    // openssl x509 -checkend returns 0 if cert is still valid, 1 if it will expire
248    if exit_code == 0 {
249        let expiry = extract_expiry_date(trimmed);
250        (
251            true,
252            format!("TLS cert for {hostname} valid for >{warn_days} days{expiry}"),
253            trimmed.to_string(),
254        )
255    } else {
256        let expiry = extract_expiry_date(trimmed);
257        (
258            false,
259            format!("TLS cert for {hostname} expires within {warn_days} days{expiry}"),
260            trimmed.to_string(),
261        )
262    }
263}
264
265fn extract_expiry_date(output: &str) -> String {
266    // Look for "notAfter=..." line
267    for line in output.lines() {
268        if let Some(date) = line.strip_prefix("notAfter=") {
269            return format!(" (expires: {date})");
270        }
271    }
272    String::new()
273}

279/// Generate probe configs from a target snapshot's listening ports and containers.
280///
281/// When `host` is provided (e.g. for SSH targets), auto-discovered probes use
282/// that address instead of `localhost`, so they hit the remote machine.
283pub fn discover_probes(snapshot: &TargetSnapshot, host: Option<&str>) -> Vec<ProbeConfig> {
284    let probe_host = host.unwrap_or("localhost");
285    let mut probes = Vec::new();
286
287    // From containers: any container exposing port 80/443 → HTTP probe
288    for c in &snapshot.containers {
289        if c.ports.contains("80") || c.ports.contains("443") {
290            let port_num = if c.ports.contains("443") { 443 } else { 80 };
291            let scheme = if port_num == 443 { "https" } else { "http" };
292            probes.push(ProbeConfig {
293                name: format!("auto-http-{}", c.name),
294                probe_type: ProbeType::Http {
295                    url: format!("{scheme}://{probe_host}:{port_num}/"),
296                    expected_status: Some(200),
297                    timeout_secs: 5,
298                },
299            });
300        }
301    }
302
303    // From listening ports
304    for p in &snapshot.listening_ports {
305        let proc_lower = p.process.to_lowercase();
306        // Any listening port with "http" or "api" in process name → HTTP probe
307        if proc_lower.contains("http") || proc_lower.contains("api") {
308            let name = format!("auto-http-{}-{}", p.process.replace('/', "-"), p.port);
309            // Avoid duplicates with container probes
310            if !probes.iter().any(|existing| existing.name == name) {
311                probes.push(ProbeConfig {
312                    name,
313                    probe_type: ProbeType::Http {
314                        url: format!("http://{probe_host}:{}/", p.port),
315                        expected_status: None,
316                        timeout_secs: 5,
317                    },
318                });
319            }
320        }
321
322        // Any 443 listener → TLS cert check
323        if p.port == 443 {
324            probes.push(ProbeConfig {
325                name: format!("auto-tls-{}", p.port),
326                probe_type: ProbeType::TlsCert {
327                    hostname: probe_host.to_string(),
328                    port: 443,
329                    warn_days: 30,
330                },
331            });
332        }
333    }
334
335    probes
336}

342/// Convert a failed probe result into an [`Alert`].
343pub fn probe_result_to_alert(result: &ProbeResult) -> Option<Alert> {
344    if result.success {
345        return None;
346    }
347
348    let (severity, category) = match result.probe_type.as_str() {
349        "tls_cert" => (AlertSeverity::Warning, AlertCategory::TlsCertExpiring),
350        "dns" => (AlertSeverity::Warning, AlertCategory::DnsResolutionFailed),
351        _ => (AlertSeverity::Critical, AlertCategory::ProbeFailure),
352    };
353
354    Some(Alert {
355        severity,
356        category,
357        message: format!("[{}] {}", result.probe_name, result.message),
358    })
359}

361/// Format probe results as a markdown summary for LLM context.
362pub fn probe_results_to_markdown(results: &[ProbeResult]) -> String {
363    use std::fmt::Write;
364    let mut md = String::new();
365    let _ = writeln!(md, "## External Probe Results\n");
366    for r in results {
367        let icon = if r.success { "OK" } else { "FAIL" };
368        let _ = writeln!(
369            md,
370            "- **[{icon}]** {name} ({ptype}): {msg}",
371            name = r.probe_name,
372            ptype = r.probe_type,
373            msg = r.message
374        );
375        if let Some(ref d) = r.details {
376            if !d.is_empty() {
377                let _ = writeln!(md, "  - Details: {d}");
378            }
379        }
380    }
381    md
382}
