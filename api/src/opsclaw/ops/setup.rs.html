24/// Expand a leading `~` to the home directory.
25fn expand_tilde(path: &str) -> String {
26    if let Some(rest) = path.strip_prefix("~/") {
27        if let Ok(home) = std::env::var("HOME") {
28            return format!("{home}/{rest}");
29        }
30    }
31    path.to_owned()
32}

44pub fn load_existing_config(path: &Path) -> OpsConfig {
45    match std::fs::read_to_string(path) {
46        Err(_) => {
47            let mut cfg = OpsConfig::default();
48            cfg.inner.config_path = path.to_path_buf();
49            cfg
50        }
51        Ok(s) => match toml::from_str::<OpsConfig>(&s) {
52            Ok(mut cfg) => {
53                cfg.inner.config_path = path.to_path_buf();
54                cfg
55            }
56            Err(e) => {
57                tracing::warn!(path = %path.display(), error = %e, "Failed to parse existing config, starting fresh");
58                let mut cfg = OpsConfig::default();
59                cfg.inner.config_path = path.to_path_buf();
60                cfg
61            }
62        },
63    }
64}

70async fn test_ssh_connection(host: &str, user: &str, port: u16, key_path: Option<&str>) -> bool {
71    let mut cmd = tokio::process::Command::new("ssh");
72    cmd.arg("-o")
73        .arg("ConnectTimeout=5")
74        .arg("-o")
75        .arg("BatchMode=yes");
76    if let Some(key) = key_path {
77        // Expand ~ to home directory
78        let expanded = if key.starts_with("~/") {
79            std::env::var("HOME")
80                .map(|h| format!("{}/{}", h, &key[2..]))
81                .unwrap_or_else(|_| key.to_string())
82        } else {
83            key.to_string()
84        };
85        cmd.arg("-i").arg(expanded);
86    }
87    let result = cmd
88        .arg("-p")
89        .arg(port.to_string())
90        .arg(format!("{user}@{host}"))
91        .arg("echo ok")
92        .output()
93        .await;
94
95    match result {
96        Ok(output) => {
97            if output.status.success() {
98                true
99            } else {
100                let stderr = String::from_utf8_lossy(&output.stderr);
101                let stdout = String::from_utf8_lossy(&output.stdout);
102                tracing::info!(
103                    host = %host,
104                    user = %user,
105                    port = port,
106                    status = ?output.status,
107                    stderr = %stderr.trim(),
108                    stdout = %stdout.trim(),
109                    "SSH connection test failed"
110                );
111                false
112            }
113        }
114        Err(e) => {
115            tracing::info!(error = %e, "Failed to spawn ssh process");
116            false
117        }
118    }
119}

144pub async fn step_ssh_target() -> Result<TargetResult> {
145    let host: String = Input::new().with_prompt("SSH host").interact_text()?;
146
147    let user: String = Input::new()
148        .with_prompt("SSH user")
149        .default("root".into())
150        .interact_text()?;
151
152    let port: u16 = Input::new()
153        .with_prompt("SSH port")
154        .default(22)
155        .interact_text()?;
156
157    let key_path: String = Input::new()
158        .with_prompt("SSH key path")
159        .default("~/.ssh/id_rsa".into())
160        .interact_text()?;
161
162    let name: String = Input::new().with_prompt("Target name").interact_text()?;
163
164    // Test connection
165    print_bullet(&format!(
166        "Testing SSH connection to {user}@{host}:{port}..."
167    ));
168
169    let mut connected = test_ssh_connection(&host, &user, port, Some(&key_path)).await;
170    if connected {
171        println!("  {} Connection successful", style("✓").green().bold());
172    } else {
173        println!("  {} Connection failed", style("✗").red().bold());
174        let retry_items = &["Retry", "Skip (continue without testing)"];
175        let choice = Select::new()
176            .with_prompt("What would you like to do?")
177            .items(retry_items)
178            .default(0)
179            .interact()?;
180        if choice == 0 {
181            connected = test_ssh_connection(&host, &user, port, Some(&key_path)).await;
182            if connected {
183                println!("  {} Connection successful", style("✓").green().bold());
184            } else {
185                println!(
186                    "  {} Connection still failing — continuing anyway",
187                    style("✗").red().bold()
188                );
189            }
190        }
191    }
192
193    // Read the SSH key content — stored inline (encrypted by Config::save()).
194    let key_pem_result = fs::read_to_string(expand_tilde(&key_path));
195
196    let runner: Box<dyn CommandRunner> = match &key_pem_result {
197        Ok(key_pem) => {
198            let entry = TargetEntry {
199                name: name.clone(),
200                host: host.clone(),
201                port,
202                user: user.clone(),
203                private_key_pem: key_pem.clone(),
204                autonomy: OpsClawAutonomy::DryRun,
205            };
206            Box::new(SshCommandRunner::new(entry, Box::new(RealSshExecutor)))
207        }
208        Err(e) => {
209            println!(
210                "  {} Could not read SSH key ({}): using stub for scan",
211                style("\u{26a0}").yellow(),
212                e
213            );
214            // Graceful fallback: scan will be skipped, not crash
215            Box::new(LocalCommandRunner::new(
216                OpsClawAutonomy::DryRun,
217                name.clone(),
218            ))
219        }
220    };
221
222    // Store key PEM content (plain text here; Config::save() encrypts it as enc2:...).
223    let key_secret = key_pem_result.ok();
224
225    let config = TargetConfig {
226        name,
227        connection_type: ConnectionType::Ssh,
228        host: Some(host),
229        port: Some(port),
230        user: Some(user),
231        key_secret,
232        autonomy: OpsClawAutonomy::default(),
233        context_file: None,
234        probes: None,
235        data_sources: None,
236        escalation: None,
237        kubeconfig: None,
238        context: None,
239        namespace: None,
240    };
241
242    Ok(TargetResult {
243        config,
244        runner: Some(runner),
245    })
246}

329/// Ask the user for optional project context and persist it to a file.
330///
331/// Returns `Some(path)` (using `~/` prefix) if context was provided, `None` otherwise.
332pub fn step_target_context(target_name: &str) -> Result<Option<String>> {
333    println!();
334    println!(
335        "  {}",
336        style("Optionally describe this target for OpsClaw (things the scan can't infer).").dim()
337    );
338    println!(
339        "  {}",
340        style("Examples: \"Redis is for sessions only\", \"batch job runs at 02:00\"").dim()
341    );
342
343    let context_input: String = Input::new()
344        .with_prompt("  Target context (Enter to skip)")
345        .allow_empty(true)
346        .interact_text()?;
347
348    if context_input.trim().is_empty() {
349        return Ok(None);
350    }
351
352    let path = write_context_file(target_name, context_input.trim())?;
353    println!(
354        "  {} Context saved to {}",
355        style("✓").green().bold(),
356        style(&path).underlined()
357    );
358    Ok(Some(path))
359}

361/// Write context content to `~/.opsclaw/context/{target_name}.md`, creating the directory
362/// if needed. Returns the path in `~/` form.
363fn write_context_file(target_name: &str, content: &str) -> Result<String> {
364    let user_dirs = directories::UserDirs::new().context("Cannot determine home directory")?;
365    let context_dir = user_dirs.home_dir().join(".opsclaw").join("context");
366    std::fs::create_dir_all(&context_dir)
367        .with_context(|| format!("Cannot create directory: {}", context_dir.display()))?;
368
369    let file_path = context_dir.join(format!("{target_name}.md"));
370    std::fs::write(&file_path, content)
371        .with_context(|| format!("Failed to write context file: {}", file_path.display()))?;
372
373    // Return tilde-prefixed path for config portability.
374    Ok(format!("~/.opsclaw/context/{target_name}.md"))
375}

395pub async fn step_discovery_scan(
396    target_name: &str,
397    runner: &dyn CommandRunner,
398) -> Option<TargetSnapshot> {
399    print_bullet(&format!("Running discovery scan on {target_name}..."));
400
401    match discovery::run_discovery_scan(runner).await {
402        Ok(snapshot) => {
403            println!("  {} Scan complete", style("✓").green().bold());
404            print_bullet(&format!(
405                "OS: {} {}",
406                snapshot.os.distro_name, snapshot.os.distro_version
407            ));
408            print_bullet(&format!(
409                "Containers: {} ({})",
410                snapshot.containers.len(),
411                snapshot
412                    .containers
413                    .iter()
414                    .map(|c| c.name.as_str())
415                    .collect::<Vec<_>>()
416                    .join(", ")
417            ));
418            print_bullet(&format!("Services: {} active", snapshot.services.len()));
419            print_bullet(&format!(
420                "Ports: {}",
421                snapshot
422                    .listening_ports
423                    .iter()
424                    .map(|p| p.port.to_string())
425                    .collect::<Vec<_>>()
426                    .join(", ")
427            ));
428            // Summarise disk from the root mount or first entry
429            if let Some(d) = snapshot.disk.first() {
430                print_bullet(&format!(
431                    "Disk: {} / {} ({}% used)",
432                    d.used, d.size, d.use_percent
433                ));
434            }
435
436            Some(snapshot)
437        }
438        Err(e) => {
439            println!("  {} Scan failed: {}", style("✗").yellow().bold(), e);
440            print_bullet("You can re-run later with: opsclaw scan");
441            None
442        }
443    }
444}

446pub fn step_data_sources() -> Result<Option<crate::ops::data_sources::DataSourcesConfig>> {
447    use crate::ops::data_sources::{DataSourcesConfig, JaegerConfig, PrometheusConfig, SeqConfig};
448
449    println!();
450    println!(
451        "  {}",
452        style("Configure endpoints for observability services (all optional).").dim()
453    );
454
455    let mut config = DataSourcesConfig::default();
456    let mut any = false;
457
458    // Prometheus
459    if Confirm::new()
460        .with_prompt("Do you have Prometheus running?")
461        .default(false)
462        .interact()?
463    {
464        let url: String = Input::new()
465            .with_prompt("Prometheus URL")
466            .default("http://localhost:9090".into())
467            .interact_text()?;
468        config.prometheus = Some(PrometheusConfig { url, token: None });
469        any = true;
470    }
471
472    // Seq
473    if Confirm::new()
474        .with_prompt("Do you have Seq running?")
475        .default(false)
476        .interact()?
477    {
478        let url: String = Input::new()
479            .with_prompt("Seq URL")
480            .default("http://localhost:5341".into())
481            .interact_text()?;
482        let api_key = crate::secrets::prompt_secret_source("the Seq API key", true)?;
483        config.seq = Some(SeqConfig { url, api_key });
484        any = true;
485    }
486
487    // Jaeger
488    if Confirm::new()
489        .with_prompt("Do you have Jaeger running?")
490        .default(false)
491        .interact()?
492    {
493        let url: String = Input::new()
494            .with_prompt("Jaeger URL")
495            .default("http://localhost:16686".into())
496            .interact_text()?;
497        config.jaeger = Some(JaegerConfig { url });
498        any = true;
499    }
500
501    if any { Ok(Some(config)) } else { Ok(None) }
502}
