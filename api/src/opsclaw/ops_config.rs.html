11/// OpsClaw configuration — wraps the upstream zeroclawlabs `Config` with
12/// SRE-specific fields (projects, notifications, diagnosis, a2a).
13#[derive(Debug, Clone, Serialize, Deserialize, Default)]
14pub struct OpsConfig {
15    /// The upstream zeroclawlabs configuration.
16    #[serde(flatten)]
17    pub inner: zeroclaw::Config,
18
19    /// SRE targets (monitored endpoints). Flat form — used when the
20    /// hierarchy is not yet adopted. Mutually exclusive with `projects`.
21    #[serde(default, skip_serializing_if = "Option::is_none")]
22    pub targets: Option<Vec<TargetConfig>>,
23
24    /// Hierarchical form: products wrapping environments wrapping targets.
25    /// Mutually exclusive with flat `targets` — see ADR-005.
26    #[serde(default, skip_serializing_if = "Vec::is_empty")]
27    pub projects: Vec<ProjectConfig>,
28
29    /// Notification delivery settings for alerts.
30    #[serde(default, skip_serializing_if = "Option::is_none")]
31    pub notifications: Option<OpsClawNotificationConfig>,
32
33    /// LLM diagnosis configuration.
34    #[serde(default)]
35    pub diagnosis: DiagnosisConfig,
36
37    /// A2A (Agent-to-Agent) protocol configuration.
38    #[serde(default, skip_serializing_if = "Option::is_none")]
39    pub a2a: Option<A2aConfig>,
40
41    /// Postgres instances for the `postgres` tool (driver-based).
42    /// Kept at the root because the driver-based pool doesn't yet have
43    /// an equivalent slot in `EndpointsConfig`.
44    #[serde(default, skip_serializing_if = "Option::is_none")]
45    pub postgres: Option<Vec<PostgresInstanceConfig>>,
46}

167impl OpsConfig {
168    /// Resolve a secret reference from config to its plaintext value.
169    ///
170    /// Accepts any of: `enc2:<hex>` (encrypted store), `enc:<hex>` (legacy
171    /// XOR, auto-upgraded on next save), `env:<NAME>` (process env var),
172    /// `k8s:<ns>/<name>/<key>` (mounted Secret volume, kube API fallback),
173    /// or a bare plaintext value (returned as-is for backward compat).
174    ///
175    /// Async because the k8s mounted-file path does real I/O; the other
176    /// backends are cheap.
177    pub async fn decrypt_secret(&self, value: &str) -> Result<String> {
178        let config_dir = self
179            .inner
180            .config_path
181            .parent()
182            .context("config path has no parent")?;
183        let composite =
184            crate::secrets::CompositeResolver::default_for(config_dir, self.inner.secrets.encrypt);
185        composite.resolve(value).await
186    }
187
188    /// Serialize the full `OpsConfig` (including opsclaw-specific fields) to
189    /// `self.config_path`. Must be called instead of `Config::save()` whenever
190    /// the caller may have modified `projects`, `notifications`, `diagnosis`, or
191    /// `a2a`, because `Config::save()` only serializes the inner zeroclaw fields
192    /// and would silently drop those fields.
193    pub async fn save(&self) -> Result<()> {
194        let config_path = &self.inner.config_path;
195        if let Some(parent) = config_path.parent() {
196            tokio::fs::create_dir_all(parent).await.with_context(|| {
197                format!("Failed to create config directory: {}", parent.display())
198            })?;
199        }
200
201        // Encrypt secret fields on targets before serializing.
202        let mut to_save = self.clone();
203        if let Some(ref mut targets) = to_save.targets {
204            let store = zeroclaw::security::SecretStore::new(
205                config_path.parent().context("config path has no parent")?,
206                self.inner.secrets.encrypt,
207            );
208            for target in targets.iter_mut() {
209                if let Some(ref val) = target.key_secret {
210                    if !crate::secrets::is_reference(val) {
211                        target.key_secret = Some(store.encrypt(val)?);
212                    }
213                }
214                if let Some(ref mut ds) = target.data_sources {
215                    if let Some(ref mut seq) = ds.seq {
216                        if let Some(ref val) = seq.api_key {
217                            if !crate::secrets::is_reference(val) {
218                                seq.api_key = Some(store.encrypt(val)?);
219                            }
220                        }
221                    }
222                    if let Some(ref mut github) = ds.github {
223                        if let Some(ref val) = github.token {
224                            if !crate::secrets::is_reference(val) {
225                                github.token = Some(store.encrypt(val)?);
226                            }
227                        }
228                    }
229                    if let Some(ref mut prometheus) = ds.prometheus {
230                        if let Some(ref val) = prometheus.token {
231                            if !crate::secrets::is_reference(val) {
232                                prometheus.token = Some(store.encrypt(val)?);
233                            }
234                        }
235                    }
236                    if let Some(ref mut es) = ds.elasticsearch {
237                        if let Some(ref val) = es.api_key {
238                            if !crate::secrets::is_reference(val) {
239                                es.api_key = Some(store.encrypt(val)?);
240                            }
241                        }
242                        if let Some(ref val) = es.password {
243                            if !crate::secrets::is_reference(val) {
244                                es.password = Some(store.encrypt(val)?);
245                            }
246                        }
247                    }
248                }
249            }
250        }
251
252        // Encrypt notification secret fields.
253        if let Some(ref mut notif) = to_save.notifications {
254            let store = zeroclaw::security::SecretStore::new(
255                config_path.parent().context("config path has no parent")?,
256                self.inner.secrets.encrypt,
257            );
258            if let Some(ref val) = notif.telegram_bot_token {
259                if !crate::secrets::is_reference(val) {
260                    notif.telegram_bot_token = Some(store.encrypt(val)?);
261                }
262            }
263            if let Some(ref val) = notif.slack_webhook_url {
264                if !crate::secrets::is_reference(val) {
265                    notif.slack_webhook_url = Some(store.encrypt(val)?);
266                }
267            }
268            if let Some(ref val) = notif.webhook_bearer_token {
269                if !crate::secrets::is_reference(val) {
270                    notif.webhook_bearer_token = Some(store.encrypt(val)?);
271                }
272            }
273        }
274
275        // Encrypt diagnosis secret fields.
276        {
277            let store = zeroclaw::security::SecretStore::new(
278                config_path.parent().context("config path has no parent")?,
279                self.inner.secrets.encrypt,
280            );
281            if let Some(ref val) = to_save.diagnosis.api_key {
282                if !crate::secrets::is_reference(val) {
283                    to_save.diagnosis.api_key = Some(store.encrypt(val)?);
284                }
285            }
286        }
287
288        // Encrypt A2A secret fields.
289        if let Some(ref mut a2a) = to_save.a2a {
290            let store = zeroclaw::security::SecretStore::new(
291                config_path.parent().context("config path has no parent")?,
292                self.inner.secrets.encrypt,
293            );
294            if !a2a.server.token.is_empty() && !crate::secrets::is_reference(&a2a.server.token) {
295                a2a.server.token = store.encrypt(&a2a.server.token)?;
296            }
297            for peer in &mut a2a.peers {
298                if !peer.token.is_empty() && !crate::secrets::is_reference(&peer.token) {
299                    peer.token = store.encrypt(&peer.token)?;
300                }
301            }
302        }
303
304        // Encrypt secret fields under projects[*].environments[*].endpoints.
305        // The legacy flat-targets pass above doesn't walk the hierarchy, so
306        // every endpoint pool — pagerduty.api_key, github.token,
307        // cloudflare.api_token, rabbitmq.password, azure_service_bus.*,
308        // posthog.api_key, prometheus[*].bearer_token, loki[*].bearer_token,
309        // elk[*].password / .api_key — needs encrypting here.
310        {
311            let store = zeroclaw::security::SecretStore::new(
312                config_path.parent().context("config path has no parent")?,
313                self.inner.secrets.encrypt,
314            );
315            for project in &mut to_save.projects {
316                for env in &mut project.environments {
317                    let eps = &mut env.endpoints;
318                    if let Some(list) = eps.prometheus.as_mut() {
319                        for e in list.iter_mut() {
320                            encrypt_in_place(&mut e.bearer_token, &store)?;
321                        }
322                    }
323                    if let Some(list) = eps.loki.as_mut() {
324                        for e in list.iter_mut() {
325                            encrypt_in_place(&mut e.bearer_token, &store)?;
326                        }
327                    }
328                    if let Some(list) = eps.elk.as_mut() {
329                        for e in list.iter_mut() {
330                            encrypt_in_place(&mut e.password, &store)?;
331                            encrypt_in_place(&mut e.api_key, &store)?;
332                        }
333                    }
334                    if let Some(pd) = eps.pagerduty.as_mut()
335                        && !crate::secrets::is_reference(&pd.api_key)
336                    {
337                        pd.api_key = store.encrypt(&pd.api_key)?;
338                    }
339                    if let Some(gh) = eps.github.as_mut()
340                        && !crate::secrets::is_reference(&gh.token)
341                    {
342                        gh.token = store.encrypt(&gh.token)?;
343                    }
344                    if let Some(cf) = eps.cloudflare.as_mut()
345                        && !crate::secrets::is_reference(&cf.api_token)
346                    {
347                        cf.api_token = store.encrypt(&cf.api_token)?;
348                    }
349                    if let Some(rmq) = eps.rabbitmq.as_mut()
350                        && !crate::secrets::is_reference(&rmq.password)
351                    {
352                        rmq.password = store.encrypt(&rmq.password)?;
353                    }
354                    if let Some(asb) = eps.azure_service_bus.as_mut()
355                        && !crate::secrets::is_reference(&asb.sas_key)
356                    {
357                        asb.sas_key = store.encrypt(&asb.sas_key)?;
358                    }
359                    if let Some(ph) = eps.posthog.as_mut()
360                        && !crate::secrets::is_reference(&ph.api_key)
361                    {
362                        ph.api_key = store.encrypt(&ph.api_key)?;
363                    }
364                }
365            }
366        }
367
368        let toml_str = toml::to_string_pretty(&to_save).context("Failed to serialize OpsConfig")?;
369        tokio::fs::write(config_path, toml_str)
370            .await
371            .with_context(|| format!("Failed to write config to {}", config_path.display()))
372    }
373}

375/// Encrypt an `Option<String>` secret in place, skipping when it's already
376/// a reference (`enc2:` / `env:` / `k8s:`) or absent.
377fn encrypt_in_place(
378    slot: &mut Option<String>,
379    store: &zeroclaw::security::SecretStore,
380) -> Result<()> {
381    if let Some(val) = slot.as_ref()
382        && !crate::secrets::is_reference(val)
383    {
384        *slot = Some(store.encrypt(val)?);
385    }
386    Ok(())
387}

389/// Three user-facing modes: `dry-run`, `approve`, `auto`.

394    DryRun,

746/// Result of resolving a `project::environment::target` address. Autonomy
747/// is already cascaded (target override > environment default > Approve).
748#[derive(Debug, Clone)]
749pub struct ResolvedTarget<'a> {
750    pub project: Option<&'a ProjectConfig>,
751    pub environment: Option<&'a EnvironmentConfig>,
752    pub target: &'a TargetConfig,
753    pub autonomy: OpsClawAutonomy,
754}
755
756impl OpsConfig {
757    /// Resolve an address (`project::environment::target`, or short forms
758    /// when unambiguous) to a `ResolvedTarget`. Works for both the flat
759    /// `targets` list and the hierarchical `projects` form.
760    ///
761    /// Short addresses:
762    /// - `target` — permitted when there is only one project and one
763    ///   environment, or the flat form is in use.
764    /// - `environment::target` — permitted when there is only one project.
765    pub fn resolve_target(&self, address: &str) -> Result<ResolvedTarget<'_>> {
766        let parts: Vec<&str> = address.split("::").collect();
767
768        if !self.projects.is_empty() {
769            match parts.as_slice() {
770                [p, e, t] => self.lookup_hier(Some(p), Some(e), t),
771                [e, t] => self.lookup_hier(None, Some(e), t),
772                [t] => self.lookup_hier(None, None, t),
773                _ => anyhow::bail!("invalid address: {address}"),
774            }
775        } else {
776            match parts.as_slice() {
777                [t] => self.lookup_flat(t),
778                _ => anyhow::bail!(
779                    "flat `targets` config accepts bare target names only; got {address}"
780                ),
781            }
782        }
783    }
784
785    fn lookup_flat(&self, name: &str) -> Result<ResolvedTarget<'_>> {
786        let targets = self.targets.as_deref().unwrap_or_default();
787        let target = targets
788            .iter()
789            .find(|t| t.name == name)
790            .ok_or_else(|| anyhow::anyhow!("target '{name}' not found"))?;
791        Ok(ResolvedTarget {
792            project: None,
793            environment: None,
794            target,
795            autonomy: target.autonomy,
796        })
797    }
798
799    fn lookup_hier(
800        &self,
801        project_name: Option<&str>,
802        env_name: Option<&str>,
803        target_name: &str,
804    ) -> Result<ResolvedTarget<'_>> {
805        let mut matches: Vec<(&ProjectConfig, &EnvironmentConfig, &TargetConfig)> = Vec::new();
806        for project in &self.projects {
807            if let Some(pn) = project_name {
808                if project.name != pn {
809                    continue;
810                }
811            }
812            for env in &project.environments {
813                if let Some(en) = env_name {
814                    if env.name != en {
815                        continue;
816                    }
817                }
818                for target in &env.targets {
819                    if target.name == target_name {
820                        matches.push((project, env, target));
821                    }
822                }
823            }
824        }
825        match matches.as_slice() {
826            [] => anyhow::bail!("no target matches '{target_name}'"),
827            [one] => {
828                let (project, environment, target) = *one;
829                let autonomy = resolve_autonomy(environment.autonomy, target.autonomy)?;
830                Ok(ResolvedTarget {
831                    project: Some(project),
832                    environment: Some(environment),
833                    target,
834                    autonomy,
835                })
836            }
837            many => {
838                let listing: Vec<String> = many
839                    .iter()
840                    .map(|(p, e, t)| format!("{}::{}::{}", p.name, e.name, t.name))
841                    .collect();
842                anyhow::bail!(
843                    "ambiguous address — {} candidates: {}",
844                    listing.len(),
845                    listing.join(", ")
846                )
847            }
848        }
849    }
850
851    /// Validate hierarchy invariants:
852    /// - Flat `[[targets]]` and hierarchical `[[projects]]` are mutually exclusive.
853    /// - Project names are unique.
854    /// - Environment names are unique within a project.
855    pub fn validate_hierarchy(&self) -> Result<()> {
856        let has_flat = self
857            .targets
858            .as_ref()
859            .map(|t| !t.is_empty())
860            .unwrap_or(false);
861        let has_hier = !self.projects.is_empty();
862        if has_flat && has_hier {
863            anyhow::bail!(
864                "config has both flat `[[targets]]` and hierarchical `[[projects]]` — \
865                 they are mutually exclusive (see ADR-005)"
866            );
867        }
868
869        // context_file paths must be relative and not traverse upward — the
870        // workspace_dir is the only legitimate root. An absolute path or `..`
871        // segment would let a malicious or buggy config read arbitrary files
872        // when the agent loads the context at run time.
873        if let Some(targets) = self.targets.as_ref() {
874            for t in targets {
875                validate_context_file(t.context_file.as_deref(), &format!("target '{}'", t.name))?;
876            }
877        }
878        let mut seen_projects = std::collections::HashSet::new();
879        for project in &self.projects {
880            if !seen_projects.insert(project.name.as_str()) {
881                anyhow::bail!("duplicate project name '{}'", project.name);
882            }
883            validate_context_file(
884                project.context_file.as_deref(),
885                &format!("project '{}'", project.name),
886            )?;
887            let mut seen_envs = std::collections::HashSet::new();
888            for env in &project.environments {
889                if !seen_envs.insert(env.name.as_str()) {
890                    anyhow::bail!(
891                        "duplicate environment name '{}' in project '{}'",
892                        env.name,
893                        project.name
894                    );
895                }
896                validate_context_file(
897                    env.context_file.as_deref(),
898                    &format!("project '{}' environment '{}'", project.name, env.name),
899                )?;
900                for t in &env.targets {
901                    validate_context_file(
902                        t.context_file.as_deref(),
903                        &format!(
904                            "project '{}' environment '{}' target '{}'",
905                            project.name, env.name, t.name
906                        ),
907                    )?;
908                }
909            }
910        }
911        Ok(())
912    }
913}
914
915/// Reject context_file paths that are absolute or contain `..` components.
916/// Empty strings and `None` are no-ops. Validation is purely lexical — we
917/// don't touch the filesystem here, the loader resolves relative to the
918/// workspace_dir later.
919fn validate_context_file(value: Option<&str>, owner: &str) -> Result<()> {
920    let Some(raw) = value else { return Ok(()) };
921    if raw.is_empty() {
922        return Ok(());
923    }
924    let path = std::path::Path::new(raw);
925    if path.is_absolute() {
926        anyhow::bail!(
927            "context_file for {owner} must be a workspace-relative path, got absolute '{raw}'"
928        );
929    }
930    for component in path.components() {
931        match component {
932            std::path::Component::ParentDir => {
933                anyhow::bail!(
934                    "context_file for {owner} contains a '..' segment; paths must stay within the workspace"
935                );
936            }
937            std::path::Component::Prefix(_) | std::path::Component::RootDir => {
938                anyhow::bail!(
939                    "context_file for {owner} must be a workspace-relative path, got '{raw}'"
940                );
941            }
942            _ => {}
943        }
944    }
945    Ok(())
946}

948/// Resolve autonomy for a target: target override must be equal to or more
949/// restrictive than the environment default. Rank: DryRun < Approve < Auto.
950fn resolve_autonomy(
951    env: Option<OpsClawAutonomy>,
952    target: OpsClawAutonomy,
953) -> Result<OpsClawAutonomy> {
954    let env_level = env.unwrap_or(OpsClawAutonomy::Approve);
955    if autonomy_rank(target) <= autonomy_rank(env_level) {
956        Ok(target)
957    } else {
958        anyhow::bail!(
959            "target autonomy '{target:?}' loosens the environment default '{env_level:?}'; \
960             overrides must be restrictive-only"
961        )
962    }
963}
964
965fn autonomy_rank(a: OpsClawAutonomy) -> u8 {
966    match a {
967        OpsClawAutonomy::DryRun => 0,
968        OpsClawAutonomy::Approve => 1,
969        OpsClawAutonomy::Auto => 2,
970    }
971}
