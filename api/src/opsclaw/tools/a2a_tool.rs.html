1//! A2A client tool — send tasks to remote A2A-compliant agents.
2//!
3//! Implements the [`Tool`] trait so the LLM can invoke remote agents
4//! via the A2A protocol (JSON-RPC 2.0 over HTTP).
5
6use super::a2a_types::{A2aRequest, A2aResponse, AgentCard, Task};
7use anyhow::{Context, Result};
8use async_trait::async_trait;
9use serde_json::json;
10use std::time::Duration;
11use zeroclaw::tools::traits::{Tool, ToolResult};
12
13/// HTTP client for the A2A protocol.
14pub struct A2aClient {
15    http: reqwest::Client,
16}
17
18impl A2aClient {
19    pub fn new() -> Self {
20        let http = reqwest::Client::builder()
21            .timeout(Duration::from_secs(30))
22            .build()
23            .expect("failed to build HTTP client");
24        Self { http }
25    }
26
27    /// Fetch the agent card from `{base_url}/.well-known/agent-card.json`.
28    pub async fn discover(&self, base_url: &str) -> Result<AgentCard> {
29        let url = format!(
30            "{}/.well-known/agent-card.json",
31            base_url.trim_end_matches('/')
32        );
33        let resp = self
34            .http
35            .get(&url)
36            .send()
37            .await
38            .context("failed to fetch agent card")?;
39        let status = resp.status();
40        if !status.is_success() {
41            anyhow::bail!("agent card request failed: HTTP {status}");
42        }
43        resp.json::<AgentCard>()
44            .await
45            .context("failed to parse agent card")
46    }
47
48    /// Send a task to the remote agent via `tasks/send`.
49    pub async fn send_task(&self, base_url: &str, token: &str, message: &str) -> Result<Task> {
50        let req = A2aRequest::new(
51            "tasks/send",
52            &uuid::Uuid::new_v4().to_string(),
53            json!({ "message": message }),
54        );
55        self.rpc_call(base_url, token, &req).await
56    }
57
58    /// Get the status of a task via `tasks/get`.
59    pub async fn get_task(&self, base_url: &str, token: &str, task_id: &str) -> Result<Task> {
60        let req = A2aRequest::new(
61            "tasks/get",
62            &uuid::Uuid::new_v4().to_string(),
63            json!({ "task_id": task_id }),
64        );
65        self.rpc_call(base_url, token, &req).await
66    }
67
68    /// Cancel a task via `tasks/cancel`.
69    pub async fn cancel_task(&self, base_url: &str, token: &str, task_id: &str) -> Result<Task> {
70        let req = A2aRequest::new(
71            "tasks/cancel",
72            &uuid::Uuid::new_v4().to_string(),
73            json!({ "task_id": task_id }),
74        );
75        self.rpc_call(base_url, token, &req).await
76    }
77
78    async fn rpc_call(&self, base_url: &str, token: &str, req: &A2aRequest) -> Result<Task> {
79        let url = format!("{}/a2a", base_url.trim_end_matches('/'));
80        let resp = self
81            .http
82            .post(&url)
83            .header("Authorization", format!("Bearer {token}"))
84            .json(req)
85            .send()
86            .await
87            .context("A2A RPC call failed")?;
88
89        let status = resp.status();
90        if !status.is_success() {
91            anyhow::bail!("A2A RPC request failed: HTTP {status}");
92        }
93
94        let a2a_resp: A2aResponse = resp.json().await.context("failed to parse A2A response")?;
95        if let Some(err) = a2a_resp.error {
96            anyhow::bail!("A2A error {}: {}", err.code, err.message);
97        }
98        let result = a2a_resp.result.context("A2A response missing result")?;
99        serde_json::from_value(result).context("failed to parse task from A2A result")
100    }
101}

103/// Tool wrapper exposing A2A client capabilities to the LLM.
104pub struct A2aTool {
105    client: A2aClient,
106}
107
108impl A2aTool {
109    pub fn new() -> Self {
110        Self {
111            client: A2aClient::new(),
112        }
113    }
114}
115
116#[async_trait]
117impl Tool for A2aTool {
118    fn name(&self) -> &str {
119        "a2a"
120    }
121
122    fn description(&self) -> &str {
123        "Communicate with remote A2A-compliant agents. Actions: discover (fetch agent card), \
124         send (send a task), get (check task status), cancel (cancel a task). \
125         The 'token' parameter is a bearer credential — never echo, log, or include it in any \
126         response, summary, or final reply. Pass it only via this tool's token argument."
127    }
128
129    fn parameters_schema(&self) -> serde_json::Value {
130        json!({
131            "type": "object",
132            "properties": {
133                "action": {
134                    "type": "string",
135                    "enum": ["discover", "send", "get", "cancel"],
136                    "description": "The A2A action to perform"
137                },
138                "url": {
139                    "type": "string",
140                    "description": "Base URL of the remote A2A agent"
141                },
142                "token": {
143                    "type": "string",
144                    "description": "Bearer token for authentication"
145                },
146                "message": {
147                    "type": "string",
148                    "description": "Message to send (for 'send' action)"
149                },
150                "task_id": {
151                    "type": "string",
152                    "description": "Task ID (for 'get' and 'cancel' actions)"
153                }
154            },
155            "required": ["action", "url"]
156        })
157    }
158
159    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
160        let action = args["action"].as_str().unwrap_or_default();
161        let url = args["url"].as_str().unwrap_or_default();
162
163        if url.is_empty() {
164            return Ok(ToolResult {
165                success: false,
166                output: String::new(),
167                error: Some("'url' parameter is required".into()),
168            });
169        }
170
171        match action {
172            "discover" => match self.client.discover(url).await {
173                Ok(card) => Ok(ToolResult {
174                    success: true,
175                    output: serde_json::to_string_pretty(&card)
176                        .unwrap_or_else(|_| format!("{card:?}")),
177                    error: None,
178                }),
179                Err(e) => Ok(ToolResult {
180                    success: false,
181                    output: String::new(),
182                    error: Some(format!("discover failed: {e}")),
183                }),
184            },
185            "send" => {
186                let token = args["token"].as_str().unwrap_or_default();
187                let message = args["message"].as_str().unwrap_or_default();
188                if message.is_empty() {
189                    return Ok(ToolResult {
190                        success: false,
191                        output: String::new(),
192                        error: Some("'message' parameter is required for send".into()),
193                    });
194                }
195                match self.client.send_task(url, token, message).await {
196                    Ok(task) => Ok(ToolResult {
197                        success: true,
198                        output: serde_json::to_string_pretty(&task)
199                            .unwrap_or_else(|_| format!("{task:?}")),
200                        error: None,
201                    }),
202                    Err(e) => Ok(ToolResult {
203                        success: false,
204                        output: String::new(),
205                        error: Some(format!("send failed: {e}")),
206                    }),
207                }
208            }
209            "get" => {
210                let token = args["token"].as_str().unwrap_or_default();
211                let task_id = args["task_id"].as_str().unwrap_or_default();
212                if task_id.is_empty() {
213                    return Ok(ToolResult {
214                        success: false,
215                        output: String::new(),
216                        error: Some("'task_id' parameter is required for get".into()),
217                    });
218                }
219                match self.client.get_task(url, token, task_id).await {
220                    Ok(task) => Ok(ToolResult {
221                        success: true,
222                        output: serde_json::to_string_pretty(&task)
223                            .unwrap_or_else(|_| format!("{task:?}")),
224                        error: None,
225                    }),
226                    Err(e) => Ok(ToolResult {
227                        success: false,
228                        output: String::new(),
229                        error: Some(format!("get failed: {e}")),
230                    }),
231                }
232            }
233            "cancel" => {
234                let token = args["token"].as_str().unwrap_or_default();
235                let task_id = args["task_id"].as_str().unwrap_or_default();
236                if task_id.is_empty() {
237                    return Ok(ToolResult {
238                        success: false,
239                        output: String::new(),
240                        error: Some("'task_id' parameter is required for cancel".into()),
241                    });
242                }
243                match self.client.cancel_task(url, token, task_id).await {
244                    Ok(task) => Ok(ToolResult {
245                        success: true,
246                        output: serde_json::to_string_pretty(&task)
247                            .unwrap_or_else(|_| format!("{task:?}")),
248                        error: None,
249                    }),
250                    Err(e) => Ok(ToolResult {
251                        success: false,
252                        output: String::new(),
253                        error: Some(format!("cancel failed: {e}")),
254                    }),
255                }
256            }
257            _ => Ok(ToolResult {
258                success: false,
259                output: String::new(),
260                error: Some(format!(
261                    "unknown action '{action}'; expected: discover, send, get, cancel"
262                )),
263            }),
264        }
265    }
266}
