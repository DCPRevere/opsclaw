1//! Discovery scan engine for mapping what is running on a target host.
2//!
3//! Runs a set of standard commands via a [`CommandRunner`] and parses their
4//! output into a [`TargetSnapshot`] that captures OS info, containers,
5//! services, listening ports, disk, memory, and load.
6
7use anyhow::Result;
8use async_trait::async_trait;
9use chrono::{DateTime, Utc};
10use serde::{Deserialize, Serialize};
11use tracing::warn;

13// ---------------------------------------------------------------------------
14// CommandRunner abstraction

17/// Output of a single command execution.
18#[derive(Debug, Clone, Serialize, Deserialize)]
19pub struct CommandOutput {
20    pub stdout: String,
21    pub stderr: String,
22    pub exit_code: i32,
23}
24
25/// Abstraction over how commands are executed (SSH, local shell, mock, …).
26#[async_trait]
27pub trait CommandRunner: Send + Sync {
28    async fn run(&self, command: &str) -> Result<CommandOutput>;
29}

163pub fn parse_uname(raw: &str) -> String {
164    raw.trim().to_string()
165}

167pub fn parse_os_release(raw: &str) -> (String, String) {
168    let mut name = String::new();
169    let mut version = String::new();
170    for line in raw.lines() {
171        let line = line.trim();
172        if let Some(val) = line.strip_prefix("NAME=") {
173            name = val.trim_matches('"').to_string();
174        } else if let Some(val) = line.strip_prefix("VERSION_ID=") {
175            version = val.trim_matches('"').to_string();
176        }
177    }
178    (name, version)
179}

181pub fn parse_ps_aux(raw: &str) -> Vec<String> {
182    raw.lines()
183        .skip(1) // header
184        .filter(|l| !l.trim().is_empty())
185        .map(|l| l.to_string())
186        .collect()
187}

189pub fn parse_ss(raw: &str) -> Vec<PortInfo> {
190    let mut ports = Vec::new();
191    for line in raw.lines().skip(1) {
192        let cols: Vec<&str> = line.split_whitespace().collect();
193        if cols.len() < 5 {
194            continue;
195        }
196        let protocol = cols[0].to_string();
197        let local_addr = cols[3];
198        // Parse address:port — handle IPv6 brackets and *:port
199        let (address, port) = split_addr_port(local_addr);
200        let process = if cols.len() >= 6 {
201            cols[5..].join(" ")
202        } else {
203            String::new()
204        };
205        ports.push(PortInfo {
206            protocol,
207            address,
208            port,
209            process,
210        });
211    }
212    ports
213}
214
215fn split_addr_port(s: &str) -> (String, u16) {
216    if let Some(idx) = s.rfind(':') {
217        let addr = &s[..idx];
218        let port_str = &s[idx + 1..];
219        let port = port_str.parse::<u16>().unwrap_or(0);
220        (addr.to_string(), port)
221    } else {
222        (s.to_string(), 0)
223    }
224}

226pub fn parse_docker_ps_json(raw: &str) -> Vec<ContainerInfo> {
227    let mut containers = Vec::new();
228    for line in raw.lines() {
229        let line = line.trim();
230        if line.is_empty() {
231            continue;
232        }
233        if let Ok(val) = serde_json::from_str::<serde_json::Value>(line) {
234            containers.push(ContainerInfo {
235                id: val["ID"].as_str().unwrap_or("").to_string(),
236                name: val["Names"].as_str().unwrap_or("").to_string(),
237                image: val["Image"].as_str().unwrap_or("").to_string(),
238                status: val["Status"].as_str().unwrap_or("").to_string(),
239                ports: val["Ports"].as_str().unwrap_or("").to_string(),
240                running_for: val["RunningFor"].as_str().unwrap_or("").to_string(),
241            });
242        }
243    }
244    containers
245}

247pub fn parse_systemctl(raw: &str) -> Vec<ServiceInfo> {
248    let mut services = Vec::new();
249    for line in raw.lines() {
250        let trimmed = line.trim();
251        // Skip header, empty lines, and legend lines
252        if trimmed.is_empty()
253            || trimmed.starts_with("UNIT")
254            || trimmed.starts_with("LOAD")
255            || trimmed.contains("loaded units listed")
256        {
257            continue;
258        }
259        let cols: Vec<&str> = trimmed.split_whitespace().collect();
260        if cols.len() < 4 {
261            continue;
262        }
263        services.push(ServiceInfo {
264            unit: cols[0].to_string(),
265            load_state: cols[1].to_string(),
266            active_state: cols[2].to_string(),
267            sub_state: cols[3].to_string(),
268            description: if cols.len() > 4 {
269                cols[4..].join(" ")
270            } else {
271                String::new()
272            },
273        });
274    }
275    services
276}

278pub fn parse_df(raw: &str) -> Vec<DiskInfo> {
279    let mut disks = Vec::new();
280    for line in raw.lines().skip(1) {
281        let cols: Vec<&str> = line.split_whitespace().collect();
282        if cols.len() < 6 {
283            continue;
284        }
285        let pct_str = cols[4].trim_end_matches('%');
286        let use_percent = pct_str.parse::<u8>().unwrap_or(0);
287        disks.push(DiskInfo {
288            filesystem: cols[0].to_string(),
289            size: cols[1].to_string(),
290            used: cols[2].to_string(),
291            available: cols[3].to_string(),
292            use_percent,
293            mount_point: cols[5].to_string(),
294        });
295    }
296    disks
297}

299pub fn parse_free(raw: &str) -> MemoryInfo {
300    for line in raw.lines() {
301        if line.starts_with("Mem:") {
302            let cols: Vec<&str> = line.split_whitespace().collect();
303            if cols.len() >= 7 {
304                return MemoryInfo {
305                    total_mb: cols[1].parse().unwrap_or(0),
306                    used_mb: cols[2].parse().unwrap_or(0),
307                    free_mb: cols[3].parse().unwrap_or(0),
308                    available_mb: cols[6].parse().unwrap_or(0),
309                };
310            }
311        }
312    }
313    MemoryInfo {
314        total_mb: 0,
315        used_mb: 0,
316        free_mb: 0,
317        available_mb: 0,
318    }
319}

321pub fn parse_uptime(raw: &str) -> LoadInfo {
322    let raw = raw.trim();
323    let (load_1, load_5, load_15) = if let Some(idx) = raw.find("load average:") {
324        let loads_str = &raw[idx + "load average:".len()..];
325        let parts: Vec<&str> = loads_str.split(',').collect();
326        let l1 = parts
327            .first()
328            .map_or(0.0, |s| s.trim().parse().unwrap_or(0.0));
329        let l5 = parts
330            .get(1)
331            .map_or(0.0, |s| s.trim().parse().unwrap_or(0.0));
332        let l15 = parts
333            .get(2)
334            .map_or(0.0, |s| s.trim().parse().unwrap_or(0.0));
335        (l1, l5, l15)
336    } else {
337        (0.0, 0.0, 0.0)
338    };
339    LoadInfo {
340        load_1,
341        load_5,
342        load_15,
343        uptime: raw.to_string(),
344    }
345}

351pub fn parse_k8s_pods_json(raw: &str) -> Vec<K8sPod> {
352    let val: serde_json::Value = match serde_json::from_str(raw) {
353        Ok(v) => v,
354        Err(e) => {
355            warn!(error = %e, "Failed to parse K8s pods JSON");
356            return Vec::new();
357        }
358    };
359    let items = match val["items"].as_array() {
360        Some(a) => a,
361        None => return Vec::new(),
362    };
363    items
364        .iter()
365        .map(|item| {
366            let metadata = &item["metadata"];
367            let status = &item["status"];
368            let phase = status["phase"].as_str().unwrap_or("");
369            // Check container statuses for more specific state (e.g. CrashLoopBackOff)
370            let container_statuses = status["containerStatuses"].as_array();
371            let (ready_str, restarts, detailed_status) = if let Some(cs) = container_statuses {
372                let total = cs.len();
373                let ready_count = cs
374                    .iter()
375                    .filter(|c| c["ready"].as_bool().unwrap_or(false))
376                    .count();
377                let restarts: u32 = cs
378                    .iter()
379                    .map(|c| {
380                        u32::try_from(c["restartCount"].as_u64().unwrap_or(0)).unwrap_or(u32::MAX)
381                    })
382                    .sum();
383                // Detect waiting reason like CrashLoopBackOff
384                let waiting_reason = cs.iter().find_map(|c| {
385                    c["state"]["waiting"]["reason"]
386                        .as_str()
387                        .map(|s| s.to_string())
388                });
389                let st = waiting_reason.unwrap_or_else(|| phase.to_string());
390                (format!("{}/{}", ready_count, total), restarts, st)
391            } else {
392                (String::from("0/0"), 0, phase.to_string())
393            };
394            let node = item["spec"]["nodeName"].as_str().unwrap_or("");
395            K8sPod {
396                name: metadata["name"].as_str().unwrap_or("").to_string(),
397                namespace: metadata["namespace"].as_str().unwrap_or("").to_string(),
398                status: detailed_status,
399                ready: ready_str,
400                restarts,
401                age: metadata["creationTimestamp"]
402                    .as_str()
403                    .unwrap_or("")
404                    .to_string(),
405                node: node.to_string(),
406            }
407        })
408        .collect()
409}

411pub fn parse_k8s_deployments_json(raw: &str) -> Vec<K8sDeployment> {
412    let val: serde_json::Value = match serde_json::from_str(raw) {
413        Ok(v) => v,
414        Err(e) => {
415            warn!(error = %e, "Failed to parse K8s deployments JSON");
416            return Vec::new();
417        }
418    };
419    let items = match val["items"].as_array() {
420        Some(a) => a,
421        None => return Vec::new(),
422    };
423    items
424        .iter()
425        .map(|item| {
426            let metadata = &item["metadata"];
427            let status = &item["status"];
428            let spec = &item["spec"];
429            let desired = u32::try_from(spec["replicas"].as_u64().unwrap_or(0)).unwrap_or(u32::MAX);
430            let available = u32::try_from(status["availableReplicas"].as_u64().unwrap_or(0))
431                .unwrap_or(u32::MAX);
432            let up_to_date =
433                u32::try_from(status["updatedReplicas"].as_u64().unwrap_or(0)).unwrap_or(u32::MAX);
434            let ready_replicas =
435                u32::try_from(status["readyReplicas"].as_u64().unwrap_or(0)).unwrap_or(u32::MAX);
436            K8sDeployment {
437                name: metadata["name"].as_str().unwrap_or("").to_string(),
438                namespace: metadata["namespace"].as_str().unwrap_or("").to_string(),
439                ready: format!("{}/{}", ready_replicas, desired),
440                up_to_date,
441                available,
442                age: metadata["creationTimestamp"]
443                    .as_str()
444                    .unwrap_or("")
445                    .to_string(),
446            }
447        })
448        .collect()
449}

451pub fn parse_k8s_services_json(raw: &str) -> Vec<K8sService> {
452    let val: serde_json::Value = match serde_json::from_str(raw) {
453        Ok(v) => v,
454        Err(e) => {
455            warn!(error = %e, "Failed to parse K8s services JSON");
456            return Vec::new();
457        }
458    };
459    let items = match val["items"].as_array() {
460        Some(a) => a,
461        None => return Vec::new(),
462    };
463    items
464        .iter()
465        .map(|item| {
466            let metadata = &item["metadata"];
467            let spec = &item["spec"];
468            let svc_type = spec["type"].as_str().unwrap_or("ClusterIP");
469            let cluster_ip = spec["clusterIP"].as_str().unwrap_or("");
470            let external_ip =
471                if let Some(ingress) = item["status"]["loadBalancer"]["ingress"].as_array() {
472                    ingress
473                        .iter()
474                        .filter_map(|i| i["ip"].as_str().or_else(|| i["hostname"].as_str()))
475                        .collect::<Vec<_>>()
476                        .join(",")
477                } else {
478                    String::from("<none>")
479                };
480            let ports = if let Some(ports_arr) = spec["ports"].as_array() {
481                ports_arr
482                    .iter()
483                    .map(|p| {
484                        let port = p["port"].as_u64().unwrap_or(0);
485                        let protocol = p["protocol"].as_str().unwrap_or("TCP");
486                        format!("{}/{}", port, protocol)
487                    })
488                    .collect::<Vec<_>>()
489                    .join(",")
490            } else {
491                String::new()
492            };
493            K8sService {
494                name: metadata["name"].as_str().unwrap_or("").to_string(),
495                namespace: metadata["namespace"].as_str().unwrap_or("").to_string(),
496                svc_type: svc_type.to_string(),
497                cluster_ip: cluster_ip.to_string(),
498                external_ip,
499                ports,
500                age: metadata["creationTimestamp"]
501                    .as_str()
502                    .unwrap_or("")
503                    .to_string(),
504            }
505        })
506        .collect()
507}

509pub fn parse_k8s_nodes_json(raw: &str) -> Vec<K8sNode> {
510    let val: serde_json::Value = match serde_json::from_str(raw) {
511        Ok(v) => v,
512        Err(e) => {
513            warn!(error = %e, "Failed to parse K8s nodes JSON");
514            return Vec::new();
515        }
516    };
517    let items = match val["items"].as_array() {
518        Some(a) => a,
519        None => return Vec::new(),
520    };
521    items
522        .iter()
523        .map(|item| {
524            let metadata = &item["metadata"];
525            let status = &item["status"];
526            let node_status = status["conditions"]
527                .as_array()
528                .and_then(|conds| {
529                    conds
530                        .iter()
531                        .find(|c| c["type"].as_str() == Some("Ready"))
532                        .map(|c| {
533                            if c["status"].as_str() == Some("True") {
534                                "Ready"
535                            } else {
536                                "NotReady"
537                            }
538                        })
539                })
540                .unwrap_or("Unknown");
541            let labels = metadata["labels"].as_object();
542            let roles = labels
543                .map(|l| {
544                    l.keys()
545                        .filter_map(|k| {
546                            k.strip_prefix("node-role.kubernetes.io/")
547                                .map(|r| r.to_string())
548                        })
549                        .collect::<Vec<_>>()
550                        .join(",")
551                })
552                .unwrap_or_default();
553            let version = status["nodeInfo"]["kubeletVersion"].as_str().unwrap_or("");
554            K8sNode {
555                name: metadata["name"].as_str().unwrap_or("").to_string(),
556                status: node_status.to_string(),
557                roles: if roles.is_empty() {
558                    "<none>".to_string()
559                } else {
560                    roles
561                },
562                age: metadata["creationTimestamp"]
563                    .as_str()
564                    .unwrap_or("")
565                    .to_string(),
566                version: version.to_string(),
567            }
568        })
569        .collect()
570}

572pub async fn discover_kubernetes(runner: &dyn CommandRunner) -> Option<KubernetesInfo> {
573    // Check if kubectl is available
574    let version_check = runner.run("kubectl version --client --short").await;
575    if version_check.is_err() {
576        return None;
577    }
578    let version_out = version_check.unwrap();
579    if version_out.exit_code != 0 {
580        return None;
581    }
582
583    let cluster_info = runner
584        .run("kubectl cluster-info 2>/dev/null")
585        .await
586        .map(|o| o.stdout.trim().to_string())
587        .unwrap_or_default();
588
589    let namespaces = runner
590        .run("kubectl get namespaces -o jsonpath='{.items[*].metadata.name}'")
591        .await
592        .map(|o| {
593            o.stdout
594                .trim()
595                .trim_matches('\'')
596                .split_whitespace()
597                .filter(|s| !s.is_empty())
598                .map(|s| s.to_string())
599                .collect()
600        })
601        .unwrap_or_default();
602
603    let pods = runner
604        .run("kubectl get pods --all-namespaces -o json")
605        .await
606        .map(|o| parse_k8s_pods_json(&o.stdout))
607        .unwrap_or_default();
608
609    let deployments = runner
610        .run("kubectl get deployments --all-namespaces -o json")
611        .await
612        .map(|o| parse_k8s_deployments_json(&o.stdout))
613        .unwrap_or_default();
614
615    let services = runner
616        .run("kubectl get services --all-namespaces -o json")
617        .await
618        .map(|o| parse_k8s_services_json(&o.stdout))
619        .unwrap_or_default();
620
621    let nodes = runner
622        .run("kubectl get nodes -o json")
623        .await
624        .map(|o| parse_k8s_nodes_json(&o.stdout))
625        .unwrap_or_default();
626
627    Some(KubernetesInfo {
628        cluster_info,
629        namespaces,
630        pods,
631        deployments,
632        services,
633        nodes,
634    })
635}

641pub async fn run_discovery_scan(runner: &dyn CommandRunner) -> Result<TargetSnapshot> {
642    let uname_out = runner.run("uname -a").await?;
643    let os_release_out = runner.run("cat /etc/os-release").await?;
644    let ss_out = runner.run("ss -tlnp").await?;
645    let systemctl_out = runner
646        .run("systemctl list-units --type=service --state=running --no-pager")
647        .await?;
648    let df_out = runner.run("df -h").await?;
649    let free_out = runner.run("free -m").await?;
650    let uptime_out = runner.run("uptime").await?;
651
652    // Docker is optional — if the command fails, treat as no containers.
653    let docker_out = runner
654        .run("docker ps --format json")
655        .await
656        .unwrap_or(CommandOutput {
657            stdout: String::new(),
658            stderr: String::new(),
659            exit_code: 1,
660        });
661
662    // Kubernetes is optional — if kubectl isn't available, skip.
663    let kubernetes = discover_kubernetes(runner).await;
664
665    let uname = parse_uname(&uname_out.stdout);
666    let (distro_name, distro_version) = parse_os_release(&os_release_out.stdout);
667
668    Ok(TargetSnapshot {
669        scanned_at: Utc::now(),
670        os: OsInfo {
671            uname,
672            distro_name,
673            distro_version,
674        },
675        containers: parse_docker_ps_json(&docker_out.stdout),
676        services: parse_systemctl(&systemctl_out.stdout),
677        listening_ports: parse_ss(&ss_out.stdout),
678        disk: parse_df(&df_out.stdout),
679        memory: parse_free(&free_out.stdout),
680        load: parse_uptime(&uptime_out.stdout),
681        kubernetes,
682    })
683}

689pub fn snapshot_to_markdown(snap: &TargetSnapshot) -> String {
690    use std::fmt::Write;
691
692    let mut md = String::new();
693    let _ = writeln!(
694        md,
695        "# Discovery Scan — {}\n",
696        snap.scanned_at.format("%Y-%m-%d %H:%M:%S UTC")
697    );
698
699    md.push_str("## OS\n\n");
700    let _ = writeln!(
701        md,
702        "- **Distro:** {} {}",
703        snap.os.distro_name, snap.os.distro_version
704    );
705    let _ = writeln!(md, "- **Kernel:** {}\n", snap.os.uname);
706
707    md.push_str("## Load\n\n");
708    let _ = writeln!(
709        md,
710        "- **Load averages:** {:.2}, {:.2}, {:.2}",
711        snap.load.load_1, snap.load.load_5, snap.load.load_15
712    );
713    let _ = writeln!(md, "- **Uptime:** {}\n", snap.load.uptime);
714
715    md.push_str("## Memory\n\n");
716    let _ = writeln!(
717        md,
718        "- **Total:** {} MB | **Used:** {} MB | **Available:** {} MB\n",
719        snap.memory.total_mb, snap.memory.used_mb, snap.memory.available_mb
720    );
721
722    md.push_str("## Disk\n\n");
723    md.push_str("| Filesystem | Size | Used | Avail | Use% | Mount |\n");
724    md.push_str("|---|---|---|---|---|---|\n");
725    for d in &snap.disk {
726        let _ = writeln!(
727            md,
728            "| {} | {} | {} | {} | {}% | {} |",
729            d.filesystem, d.size, d.used, d.available, d.use_percent, d.mount_point
730        );
731    }
732    md.push('\n');
733
734    if !snap.containers.is_empty() {
735        md.push_str("## Containers\n\n");
736        md.push_str("| Name | Image | Status | Ports |\n");
737        md.push_str("|---|---|---|---|\n");
738        for c in &snap.containers {
739            let _ = writeln!(
740                md,
741                "| {} | {} | {} | {} |",
742                c.name, c.image, c.status, c.ports
743            );
744        }
745        md.push('\n');
746    }
747
748    if !snap.services.is_empty() {
749        md.push_str("## Services\n\n");
750        md.push_str("| Unit | State | Description |\n");
751        md.push_str("|---|---|---|\n");
752        for s in &snap.services {
753            let _ = writeln!(
754                md,
755                "| {} | {} | {} |",
756                s.unit, s.active_state, s.description
757            );
758        }
759        md.push('\n');
760    }
761
762    if !snap.listening_ports.is_empty() {
763        md.push_str("## Listening Ports\n\n");
764        md.push_str("| Proto | Address | Port | Process |\n");
765        md.push_str("|---|---|---|---|\n");
766        for p in &snap.listening_ports {
767            let _ = writeln!(
768                md,
769                "| {} | {} | {} | {} |",
770                p.protocol, p.address, p.port, p.process
771            );
772        }
773        md.push('\n');
774    }
775
776    if let Some(k8s) = &snap.kubernetes {
777        md.push_str("## Kubernetes\n\n");
778        if !k8s.cluster_info.is_empty() {
779            let _ = writeln!(md, "**Cluster:** {}\n", k8s.cluster_info);
780        }
781
782        if !k8s.pods.is_empty() {
783            md.push_str("### Pods\n\n");
784            md.push_str("| Name | Namespace | Status | Ready | Restarts |\n");
785            md.push_str("|---|---|---|---|---|\n");
786            for p in &k8s.pods {
787                let highlight = if p.status.contains("CrashLoopBackOff")
788                    || p.status == "Error"
789                    || p.status == "NotReady"
790                {
791                    " **UNHEALTHY**"
792                } else {
793                    ""
794                };
795                let _ = writeln!(
796                    md,
797                    "| {} | {} | {}{} | {} | {} |",
798                    p.name, p.namespace, p.status, highlight, p.ready, p.restarts
799                );
800            }
801            md.push('\n');
802        }
803
804        if !k8s.deployments.is_empty() {
805            md.push_str("### Deployments\n\n");
806            md.push_str("| Name | Namespace | Ready |\n");
807            md.push_str("|---|---|---|\n");
808            for d in &k8s.deployments {
809                let _ = writeln!(md, "| {} | {} | {} |", d.name, d.namespace, d.ready);
810            }
811            md.push('\n');
812        }
813
814        if !k8s.nodes.is_empty() {
815            md.push_str("### Nodes\n\n");
816            md.push_str("| Name | Status | Roles | Version |\n");
817            md.push_str("|---|---|---|---|\n");
818            for n in &k8s.nodes {
819                let highlight = if n.status == "Ready" {
820                    ""
821                } else {
822                    " **UNHEALTHY**"
823                };
824                let _ = writeln!(
825                    md,
826                    "| {} | {}{} | {} | {} |",
827                    n.name, n.status, highlight, n.roles, n.version
828                );
829            }
830            md.push('\n');
831        }
832    }
833
834    md
835}
