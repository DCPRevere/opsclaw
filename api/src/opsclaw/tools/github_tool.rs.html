1//! GitHub tool. One tool, one token, many actions:
2//! issues, PRs, workflow runs, deployments, releases, commits.
3//!
4//! Writes (comment, create, rerun, cancel) are gated by autonomy and
5//! audit-logged.

47impl GithubTool {
48    pub fn new(config: GithubToolConfig) -> Self {
49        Self {
50            config,
51            client: reqwest::Client::builder()
52                .timeout(Duration::from_secs(30))
53                .user_agent("opsclaw/0.6")
54                .build()
55                .unwrap_or_else(|_| reqwest::Client::new()),
56            audit_dir: None,
57        }
58    }
59
60    pub fn with_audit_dir(mut self, dir: PathBuf) -> Self {
61        self.audit_dir = Some(dir);
62        self
63    }
64
65    fn owner_repo(&self, args: &Value) -> Result<(String, String), String> {
66        let owner = args
67            .get("owner")
68            .and_then(|v| v.as_str())
69            .map(String::from)
70            .or_else(|| self.config.default_owner.clone())
71            .ok_or_else(|| "missing 'owner' (no default_owner in config)".to_string())?;
72        let repo = args
73            .get("repo")
74            .and_then(|v| v.as_str())
75            .map(String::from)
76            .or_else(|| self.config.default_repo.clone())
77            .ok_or_else(|| "missing 'repo' (no default_repo in config)".to_string())?;
78        Ok((owner, repo))
79    }
80
81    fn req(&self, method: reqwest::Method, path: &str) -> reqwest::RequestBuilder {
82        let url = format!("{}/{}", self.config.api_base.trim_end_matches('/'), path);
83        self.client
84            .request(method, &url)
85            .header("Authorization", format!("Bearer {}", self.config.token))
86            .header("Accept", "application/vnd.github+json")
87            .header("X-GitHub-Api-Version", "2022-11-28")
88    }
89
90    fn audit(&self, action: &str, detail: &str, duration_ms: u128, exit: i32) {
91        let _ = write_audit_entry(
92            "github",
93            &format!("{action} {detail}"),
94            exit,
95            duration_ms,
96            self.audit_dir.as_ref(),
97        );
98    }
99
100    fn is_dry_run(&self) -> bool {
101        self.config.autonomy == OpsClawAutonomy::DryRun
102    }
103}

106impl Tool for GithubTool {
107    fn name(&self) -> &str {
108        "github"
109    }

111    fn description(&self) -> &str {
112        "GitHub API tool. Reads: list_issues, get_issue, list_prs, get_pr, \
113         list_pr_comments, list_runs, get_run, get_run_logs_url, \
114         list_deployments, list_releases, list_commits. Writes: \
115         create_issue, comment_issue, comment_pr, rerun_run, cancel_run. \
116         Writes respect autonomy (DryRun returns 'would do X') and are \
117         audit-logged."
118    }

120    fn parameters_schema(&self) -> Value {
121        json!({
122            "type": "object",
123            "properties": {
124                "action": {"type": "string"},
125                "owner": {"type": "string"},
126                "repo": {"type": "string"},
127                "issue_number": {"type": "integer"},
128                "pr_number": {"type": "integer"},
129                "run_id": {"type": "integer"},
130                "deployment_id": {"type": "integer"},
131                "release_id": {"type": "integer"},
132                "branch": {"type": "string"},
133                "state": {"type": "string", "enum": ["open", "closed", "all"]},
134                "status": {"type": "string"},
135                "title": {"type": "string"},
136                "body": {"type": "string"},
137                "labels": {"type": "array", "items": {"type": "string"}},
138                "per_page": {"type": "integer", "default": 30}
139            },
140            "required": ["action"]
141        })
142    }
143
144    async fn execute(&self, args: Value) -> anyhow::Result<ToolResult> {
145        let action = match args.get("action").and_then(|v| v.as_str()) {
146            Some(s) => s.to_string(),
147            None => {
148                return Ok(ToolResult {
149                    success: false,
150                    output: String::new(),
151                    error: Some("missing 'action'".into()),
152                });
153            }
154        };
155        let start = std::time::Instant::now();
156        let result = self.dispatch(&action, &args).await;
157        let elapsed = start.elapsed().as_millis();
158        let exit = match &result {
159            Ok(r) if r.success => 0,
160            _ => 1,
161        };
162        let detail = args
163            .get("issue_number")
164            .or_else(|| args.get("pr_number"))
165            .or_else(|| args.get("run_id"))
166            .map(|v| v.to_string())
167            .unwrap_or_default();
168        self.audit(&action, &detail, elapsed, exit);
169        result
170    }
171}

174    async fn dispatch(&self, action: &str, args: &Value) -> anyhow::Result<ToolResult> {
175        match action {
176            "list_issues" => self.list_issues(args).await,
177            "get_issue" => self.get_issue(args).await,
178            "list_prs" => self.list_prs(args).await,
179            "get_pr" => self.get_pr(args).await,
180            "list_pr_comments" => self.list_pr_comments(args).await,
181            "list_runs" => self.list_runs(args).await,
182            "get_run" => self.get_run(args).await,
183            "get_run_logs_url" => self.get_run_logs_url(args).await,
184            "list_deployments" => self.list_deployments(args).await,
185            "list_releases" => self.list_releases(args).await,
186            "list_commits" => self.list_commits(args).await,
187            "create_issue" => self.create_issue(args).await,
188            "comment_issue" => self.comment(args, /*is_pr=*/ false).await,
189            "comment_pr" => self.comment(args, true).await,
190            "rerun_run" => self.run_mutation(args, "rerun").await,
191            "cancel_run" => self.run_mutation(args, "cancel").await,
192            other => Ok(err(format!("unknown action '{other}'"))),
193        }
194    }

196    async fn list_issues(&self, args: &Value) -> anyhow::Result<ToolResult> {
197        let (owner, repo) = match self.owner_repo(args) {
198            Ok(p) => p,
199            Err(e) => return Ok(err(e)),
200        };
201        let state = args.get("state").and_then(|v| v.as_str()).unwrap_or("open");
202        let per_page = args
203            .get("per_page")
204            .and_then(|v| v.as_u64())
205            .unwrap_or(30)
206            .min(100);
207        let path = format!("repos/{owner}/{repo}/issues");
208        let resp = self
209            .req(reqwest::Method::GET, &path)
210            .query(&[("state", state), ("per_page", &per_page.to_string())])
211            .send()
212            .await?;
213        let (ok, status, body) = consume(resp).await;
214        if !ok {
215            return Ok(err(format!("{status}: {}", snippet(&body))));
216        }
217        let arr: Value = serde_json::from_str(&body).unwrap_or(Value::Null);
218        let mut out = String::new();
219        if let Some(a) = arr.as_array() {
220            writeln!(out, "count: {}", a.len()).ok();
221            for i in a {
222                // Skip pull requests (issues API returns both).
223                if i.get("pull_request").is_some() {
224                    continue;
225                }
226                let num = i.get("number").and_then(|v| v.as_u64()).unwrap_or(0);
227                let st = i.get("state").and_then(|v| v.as_str()).unwrap_or("");
228                let title = i.get("title").and_then(|v| v.as_str()).unwrap_or("");
229                let user = i
230                    .get("user")
231                    .and_then(|v| v.get("login"))
232                    .and_then(|v| v.as_str())
233                    .unwrap_or("");
234                writeln!(out, "  #{num} [{st}] @{user} — {title}").ok();
235            }
236        }
237        Ok(ok_res(out))
238    }

441    async fn get_run_logs_url(&self, args: &Value) -> anyhow::Result<ToolResult> {
442        let (owner, repo) = match self.owner_repo(args) {
443            Ok(p) => p,
444            Err(e) => return Ok(err(e)),
445        };
446        let id = match args.get("run_id").and_then(|v| v.as_u64()) {
447            Some(n) => n,
448            None => return Ok(err("missing 'run_id'")),
449        };
450        // GitHub returns 302 to a signed S3 URL. We don't follow it — the agent just needs the URL.
451        let path = format!("repos/{owner}/{repo}/actions/runs/{id}/logs");
452        let client = reqwest::Client::builder()
453            .redirect(reqwest::redirect::Policy::none())
454            .timeout(Duration::from_secs(30))
455            .user_agent("opsclaw/0.6")
456            .build()?;
457        let url = format!("{}/{}", self.config.api_base.trim_end_matches('/'), path);
458        let resp = client
459            .get(&url)
460            .header("Authorization", format!("Bearer {}", self.config.token))
461            .header("Accept", "application/vnd.github+json")
462            .header("X-GitHub-Api-Version", "2022-11-28")
463            .send()
464            .await?;
465        let status = resp.status();
466        if status.as_u16() == 302 || status == reqwest::StatusCode::FOUND {
467            if let Some(loc) = resp.headers().get("location").and_then(|v| v.to_str().ok()) {
468                return Ok(ok_res(format!("logs_url: {loc}")));
469            }
470        }
471        let body = resp.text().await.unwrap_or_default();
472        if status.is_success() {
473            return Ok(ok_res(format!(
474                "status: {status} (no redirect; logs inline, {} bytes)",
475                body.len()
476            )));
477        }
478        Ok(err(format!("{status}: {}", snippet(&body))))
479    }

562        let mut params: Vec<(&str, String)> = vec![("per_page", per_page.to_string())];
563        if let Some(b) = branch {
564            params.push(("sha", b.to_string()));
565        }
566        let resp = self
567            .req(reqwest::Method::GET, &path)
568            .query(&params)
569            .send()
570            .await?;
571        let (ok, status, body) = consume(resp).await;
572        if !ok {
573            return Ok(err(format!("{status}: {}", snippet(&body))));
574        }
575        let arr: Value = serde_json::from_str(&body).unwrap_or(Value::Null);
576        let mut out = String::new();
577        if let Some(a) = arr.as_array() {
578            for c in a {
579                let sha = c.get("sha").and_then(|v| v.as_str()).unwrap_or("");
580                let sha_short: String = sha.chars().take(8).collect();
581                let msg = c
582                    .get("commit")
583                    .and_then(|v| v.get("message"))
584                    .and_then(|v| v.as_str())
585                    .unwrap_or("");
586                let first = msg.lines().next().unwrap_or("");
587                let author = c
588                    .get("commit")
589                    .and_then(|v| v.get("author"))
590                    .and_then(|v| v.get("name"))
591                    .and_then(|v| v.as_str())
592                    .unwrap_or("");
593                let date = c
594                    .get("commit")
595                    .and_then(|v| v.get("author"))
596                    .and_then(|v| v.get("date"))
597                    .and_then(|v| v.as_str())
598                    .unwrap_or("");
599                writeln!(out, "  {sha_short} {date} {author}: {first}").ok();
600            }
601        }
602        Ok(ok_res(out))
603    }
604
605    async fn create_issue(&self, args: &Value) -> anyhow::Result<ToolResult> {
606        let (owner, repo) = match self.owner_repo(args) {
607            Ok(p) => p,
608            Err(e) => return Ok(err(e)),
609        };
610        let title = match args.get("title").and_then(|v| v.as_str()) {
611            Some(s) if !s.is_empty() => s.to_string(),
612            _ => return Ok(err("missing 'title'")),
613        };
614        let body = args
615            .get("body")
616            .and_then(|v| v.as_str())
617            .unwrap_or("")
618            .to_string();
619        let labels = args
620            .get("labels")
621            .and_then(|v| v.as_array())
622            .map(|a| {
623                a.iter()
624                    .filter_map(|v| v.as_str().map(String::from))
625                    .collect::<Vec<_>>()
626            })
627            .unwrap_or_default();
628
629        if self.is_dry_run() {
630            return Ok(ok_res(format!(
631                "[dry-run] would create issue in {owner}/{repo}: {title}"
632            )));
633        }
634
635        let payload = json!({"title": title, "body": body, "labels": labels});
636        let path = format!("repos/{owner}/{repo}/issues");
637        let resp = self
638            .req(reqwest::Method::POST, &path)
639            .json(&payload)
640            .send()
641            .await?;
642        let (ok, status, body) = consume(resp).await;
643        if !ok {
644            return Ok(err(format!("{status}: {}", snippet(&body))));
645        }
646        let v: Value = serde_json::from_str(&body).unwrap_or(Value::Null);
647        let num = v.get("number").and_then(|v| v.as_u64()).unwrap_or(0);
648        let url = v.get("html_url").and_then(|v| v.as_str()).unwrap_or("");
649        Ok(ok_res(format!("created issue #{num}: {url}")))
650    }

652    async fn comment(&self, args: &Value, _is_pr: bool) -> anyhow::Result<ToolResult> {
653        let (owner, repo) = match self.owner_repo(args) {
654            Ok(p) => p,
655            Err(e) => return Ok(err(e)),
656        };
657        // PR comments use the same /issues/{n}/comments endpoint.
658        let n = args
659            .get("pr_number")
660            .and_then(|v| v.as_u64())
661            .or_else(|| args.get("issue_number").and_then(|v| v.as_u64()));
662        let n = match n {
663            Some(n) => n,
664            None => return Ok(err("missing 'issue_number' or 'pr_number'")),
665        };
666        let body = match args.get("body").and_then(|v| v.as_str()) {
667            Some(s) if !s.is_empty() => s.to_string(),
668            _ => return Ok(err("missing 'body'")),
669        };
670        if self.is_dry_run() {
671            return Ok(ok_res(format!(
672                "[dry-run] would comment on {owner}/{repo}#{n}: {body}"
673            )));
674        }
675        let path = format!("repos/{owner}/{repo}/issues/{n}/comments");
676        let resp = self
677            .req(reqwest::Method::POST, &path)
678            .json(&json!({"body": body}))
679            .send()
680            .await?;
681        let (ok, status, resp_body) = consume(resp).await;
682        if !ok {
683            return Ok(err(format!("{status}: {}", snippet(&resp_body))));
684        }
685        Ok(ok_res(format!("comment added to {owner}/{repo}#{n}")))
686    }
687
688    async fn run_mutation(&self, args: &Value, op: &str) -> anyhow::Result<ToolResult> {
689        let (owner, repo) = match self.owner_repo(args) {
690            Ok(p) => p,
691            Err(e) => return Ok(err(e)),
692        };
693        let id = match args.get("run_id").and_then(|v| v.as_u64()) {
694            Some(n) => n,
695            None => return Ok(err("missing 'run_id'")),
696        };
697        if self.is_dry_run() {
698            return Ok(ok_res(format!(
699                "[dry-run] would {op} run {id} in {owner}/{repo}"
700            )));
701        }
702        let suffix = if op == "rerun" { "rerun" } else { "cancel" };
703        let path = format!("repos/{owner}/{repo}/actions/runs/{id}/{suffix}");
704        let resp = self.req(reqwest::Method::POST, &path).send().await?;
705        let (ok, status, body) = consume(resp).await;
706        if !ok {
707            return Ok(err(format!("{status}: {}", snippet(&body))));
708        }
709        Ok(ok_res(format!("{op} requested for run {id}")))
710    }
711}
712
713async fn consume(resp: reqwest::Response) -> (bool, reqwest::StatusCode, String) {
714    let status = resp.status();
715    let text = resp.text().await.unwrap_or_default();
716    (status.is_success(), status, text)
717}
718
719fn snippet(s: &str) -> &str {
720    &s[..s.len().min(500)]
721}
722
723fn err<S: Into<String>>(msg: S) -> ToolResult {
724    ToolResult {
725        success: false,
726        output: String::new(),
727        error: Some(msg.into()),
728    }
729}
730
731fn ok_res(mut s: String) -> ToolResult {
732    if s.len() > MAX_OUTPUT_BYTES {
733        let mut cut = MAX_OUTPUT_BYTES;
734        while cut > 0 && !s.is_char_boundary(cut) {
735            cut -= 1;
736        }
737        s.truncate(cut);
738        s.push_str("\n... [truncated]");
739    }
740    ToolResult {
741        success: true,
742        output: s,
743        error: None,
744    }
745}
746
747fn format_issue(v: &Value) -> String {
748    let mut out = String::new();
749    for k in [
750        "number",
751        "state",
752        "title",
753        "created_at",
754        "updated_at",
755        "html_url",
756    ] {
757        if let Some(val) = v.get(k) {
758            writeln!(out, "{k}: {val}").ok();
759        }
760    }
761    if let Some(user) = v
762        .get("user")
763        .and_then(|v| v.get("login"))
764        .and_then(|v| v.as_str())
765    {
766        writeln!(out, "user: {user}").ok();
767    }
768    if let Some(body) = v.get("body").and_then(|v| v.as_str()) {
769        let first: String = body.chars().take(1000).collect();
770        writeln!(out, "body:\n{first}").ok();
771    }
772    out
773}
