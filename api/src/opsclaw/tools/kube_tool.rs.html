36/// Abstraction over how a [`KubeClient`] is constructed so that tests can
37/// substitute a fake.
38#[async_trait::async_trait]
39pub trait KubeClientFactory: Send + Sync {
40    async fn build(&self, kubeconfig: Option<&str>, context: Option<&str>) -> Result<KubeClient>;
41}
42
43/// Default factory that creates a real client from the environment or a
44/// specific kubeconfig path.
45pub struct DefaultKubeClientFactory;
46
47#[async_trait::async_trait]
48impl KubeClientFactory for DefaultKubeClientFactory {
49    async fn build(&self, kubeconfig: Option<&str>, context: Option<&str>) -> Result<KubeClient> {
50        KubeClient::new(kubeconfig, context).await
51    }
52}

54impl KubeClient {
55    /// Create a new client.
56    ///
57    /// If `kubeconfig` is `Some`, that path is used; otherwise the default
58    /// kubeconfig resolution order applies (~/.kube/config, in-cluster).
59    /// If `context` is `Some`, that kubeconfig context is selected; otherwise
60    /// the kubeconfig's `current-context` applies.
61    pub async fn new(kubeconfig: Option<&str>, context: Option<&str>) -> Result<Self> {
62        let options = KubeConfigOptions {
63            context: context.map(String::from),
64            ..Default::default()
65        };
66        let config = match kubeconfig {
67            Some(path) => {
68                let kubeconfig = kube::config::Kubeconfig::read_from(path)
69                    .context("failed to read kubeconfig")?;
70                kube::Config::from_custom_kubeconfig(kubeconfig, &options)
71                    .await
72                    .context("failed to build kube config")?
73            }
74            None => kube::Config::infer()
75                .await
76                .context("failed to infer kube config")?,
77        };
78        let client = Client::try_from(config).context("failed to build kube client")?;
79        Ok(Self { client })
80    }

86    /// Perform a full Kubernetes discovery scan and return a
87    /// [`KubernetesInfo`] snapshot.
88    pub async fn discover_k8s(&self) -> Result<KubernetesInfo> {
89        debug!("Starting Kubernetes discovery");
90        // cluster_version is informational (may require privileges old RBAC
91        // rejects); the rest of the snapshot is meaningful without it.
92        let cluster_info = match self.cluster_version().await {
93            Ok(v) => v,
94            Err(e) => {
95                error!(error = %e, "Failed to retrieve cluster version — continuing without it");
96                String::new()
97            }
98        };
99        let namespaces = self.list_namespaces().await.context("list namespaces")?;
100        let pods = self.list_pods(None).await.context("list pods")?;
101        let deployments = self
102            .list_deployments(None)
103            .await
104            .context("list deployments")?;
105        let services = self.list_services(None).await.context("list services")?;
106        let nodes = self.list_nodes().await.context("list nodes")?;
107        debug!(
108            pods = pods.len(),
109            deployments = deployments.len(),
110            services = services.len(),
111            nodes = nodes.len(),
112            "Kubernetes discovery complete"
113        );
114
115        Ok(KubernetesInfo {
116            cluster_info,
117            namespaces,
118            pods,
119            deployments,
120            services,
121            nodes,
122        })
123    }

125    /// Build a minimal [`TargetSnapshot`] populated only with Kubernetes data.
126    pub async fn discover_snapshot(&self) -> Result<TargetSnapshot> {
127        let k8s = self.discover_k8s().await?;
128        Ok(TargetSnapshot {
129            scanned_at: Utc::now(),
130            os: super::discovery::OsInfo {
131                uname: String::new(),
132                distro_name: String::from("kubernetes"),
133                distro_version: k8s.cluster_info.clone(),
134            },
135            containers: Vec::new(),
136            services: Vec::new(),
137            listening_ports: Vec::new(),
138            disk: Vec::new(),
139            memory: super::discovery::MemoryInfo {
140                total_mb: 0,
141                used_mb: 0,
142                free_mb: 0,
143                available_mb: 0,
144            },
145            load: super::discovery::LoadInfo {
146                load_1: 0.0,
147                load_5: 0.0,
148                load_15: 0.0,
149                uptime: String::new(),

151            kubernetes: Some(k8s),

159    /// Retrieve the last `lines` of log output from a pod.
160    pub async fn get_pod_logs(&self, namespace: &str, pod: &str, lines: i64) -> Result<String> {
161        let pods: Api<Pod> = Api::namespaced(self.client.clone(), namespace);
162        let lp = LogParams {
163            tail_lines: Some(lines),
164            ..Default::default()
165        };
166        pods.logs(pod, &lp)
167            .await
168            .with_context(|| format!("failed to get logs for {namespace}/{pod}"))
169    }

175    /// Restart a deployment by annotating its pod template (the standard
176    /// `kubectl rollout restart` approach).
177    pub async fn restart_deployment(&self, namespace: &str, name: &str) -> Result<()> {
178        let deployments: Api<Deployment> = Api::namespaced(self.client.clone(), namespace);
179        let restart_annotation = serde_json::json!({
180            "spec": {
181                "template": {
182                    "metadata": {
183                        "annotations": {
184                            "kubectl.kubernetes.io/restartedAt": Utc::now().to_rfc3339()
185                        }
186                    }
187                }
188            }
189        });
190        deployments
191            .patch(
192                name,
193                &PatchParams::apply("opsclaw"),
194                &Patch::Strategic(restart_annotation),
195            )
196            .await
197            .with_context(|| format!("failed to restart deployment {namespace}/{name}"))?;
198        Ok(())
199    }

201    /// Scale a deployment to `replicas` by patching its `spec.replicas`.
202    pub async fn scale_deployment(&self, namespace: &str, name: &str, replicas: i32) -> Result<()> {
203        let deployments: Api<Deployment> = Api::namespaced(self.client.clone(), namespace);
204        let patch = serde_json::json!({ "spec": { "replicas": replicas } });
205        deployments
206            .patch(name, &PatchParams::default(), &Patch::Merge(patch))
207            .await
208            .with_context(|| format!("failed to scale {namespace}/{name} to {replicas}"))?;
209        Ok(())
210    }

218        Ok(())

225    /// Return recent events for a namespace (or all namespaces if `namespace`
226    /// is `None`).
227    pub async fn get_events(&self, namespace: Option<&str>) -> Result<Vec<K8sEvent>> {
228        let events: Api<Event> = match namespace {
229            Some(ns) => Api::namespaced(self.client.clone(), ns),
230            None => Api::all(self.client.clone()),
231        };
232        let list = events
233            .list(&ListParams::default())
234            .await
235            .context("failed to list events")?;
236
237        Ok(list
238            .items
239            .into_iter()
240            .map(|e| {
241                let obj = e.involved_object;
242                let obj_str = format!(
243                    "{}/{}",
244                    obj.kind.as_deref().unwrap_or("Unknown"),
245                    obj.name.as_deref().unwrap_or("unknown"),
246                );
247                K8sEvent {
248                    reason: e.reason.unwrap_or_default(),
249                    message: e.message.unwrap_or_default(),
250                    involved_object: obj_str,
251                    timestamp: e
252                        .last_timestamp
253                        .map(|t| t.0.to_rfc3339())
254                        .unwrap_or_default(),
255                }
256            })
257            .collect())
258    }

320fn pod_to_snapshot(pod: Pod) -> K8sPod {
321    let meta = pod.metadata;
322    let status = pod.status.unwrap_or_default();
323    let phase = status
324        .phase
325        .clone()
326        .unwrap_or_else(|| "Unknown".to_string());
327
328    let container_statuses = status.container_statuses.unwrap_or_default();
329    let total = container_statuses.len();
330    let ready_count = container_statuses.iter().filter(|c| c.ready).count();
331    let restarts: u32 = container_statuses
332        .iter()
333        .map(|c| c.restart_count.max(0).cast_unsigned())
334        .sum();
335
336    let waiting_reason = container_statuses.iter().find_map(|c| {
337        c.state
338            .as_ref()
339            .and_then(|s| s.waiting.as_ref().and_then(|w| w.reason.clone()))
340    });
341
342    let detailed_status = waiting_reason.unwrap_or(phase);
343    let node = pod.spec.and_then(|s| s.node_name).unwrap_or_default();
344
345    K8sPod {
346        name: meta.name.unwrap_or_default(),
347        namespace: meta.namespace.unwrap_or_default(),
348        status: detailed_status,
349        ready: format!("{ready_count}/{total}"),
350        restarts,
351        age: meta
352            .creation_timestamp
353            .map(|t| t.0.to_rfc3339())
354            .unwrap_or_default(),
355        node,
356    }
357}

359fn deployment_to_snapshot(dep: Deployment) -> K8sDeployment {
360    let meta = dep.metadata;
361    let spec = dep.spec.unwrap_or_default();
362    let status = dep.status.unwrap_or_default();
363    let desired = spec.replicas.unwrap_or(0);
364    let ready_replicas = status.ready_replicas.unwrap_or(0);
365    let up_to_date = status.updated_replicas.unwrap_or(0).max(0).cast_unsigned();
366    let available = status
367        .available_replicas
368        .unwrap_or(0)
369        .max(0)
370        .cast_unsigned();
371
372    K8sDeployment {
373        name: meta.name.unwrap_or_default(),
374        namespace: meta.namespace.unwrap_or_default(),
375        ready: format!("{ready_replicas}/{desired}"),
376        up_to_date,
377        available,
378        age: meta
379            .creation_timestamp
380            .map(|t| t.0.to_rfc3339())
381            .unwrap_or_default(),
382    }
383}

385fn service_to_snapshot(svc: Service) -> K8sService {
386    let meta = svc.metadata;
387    let spec = svc.spec.unwrap_or_default();
388    let svc_type = spec
389        .type_
390        .clone()
391        .unwrap_or_else(|| "ClusterIP".to_string());
392    let cluster_ip = spec.cluster_ip.clone().unwrap_or_default();
393
394    let lb = svc.status.and_then(|s| s.load_balancer).unwrap_or_default();
395    let external_ip = lb
396        .ingress
397        .unwrap_or_default()
398        .iter()
399        .filter_map(|i| i.ip.as_deref().or(i.hostname.as_deref()))
400        .collect::<Vec<_>>()
401        .join(",");
402    let external_ip = if external_ip.is_empty() {
403        "<none>".to_string()
404    } else {
405        external_ip
406    };
407
408    let ports = spec
409        .ports
410        .unwrap_or_default()
411        .iter()
412        .map(|p| format!("{}/{}", p.port, p.protocol.as_deref().unwrap_or("TCP")))
413        .collect::<Vec<_>>()
414        .join(",");
415
416    K8sService {
417        name: meta.name.unwrap_or_default(),
418        namespace: meta.namespace.unwrap_or_default(),
419        svc_type,
420        cluster_ip,
421        external_ip,
422        ports,
423        age: meta
424            .creation_timestamp
425            .map(|t| t.0.to_rfc3339())
426            .unwrap_or_default(),
427    }
428}

430fn node_to_snapshot(node: Node) -> K8sNode {
431    let meta = node.metadata;
432    let status = node.status.unwrap_or_default();
433
434    let node_status = status
435        .conditions
436        .as_ref()
437        .and_then(|conds| {
438            conds.iter().find(|c| c.type_ == "Ready").map(|c| {
439                if c.status == "True" {
440                    "Ready"
441                } else {
442                    "NotReady"
443                }
444            })
445        })
446        .unwrap_or("Unknown");
447
448    let labels = meta.labels.clone().unwrap_or_default();
449    let roles: Vec<String> = labels
450        .keys()
451        .filter_map(|k| {
452            k.strip_prefix("node-role.kubernetes.io/")
453                .map(|r| r.to_string())
454        })
455        .collect();
456    let roles = if roles.is_empty() {
457        "<none>".to_string()
458    } else {
459        roles.join(",")
460    };
461
462    let version = status
463        .node_info
464        .map(|ni| ni.kubelet_version)
465        .unwrap_or_default();
466
467    K8sNode {
468        name: meta.name.unwrap_or_default(),
469        status: node_status.to_string(),
470        roles,
471        age: meta
472            .creation_timestamp
473            .map(|t| t.0.to_rfc3339())
474            .unwrap_or_default(),
475        version,
476    }
477}

693    pub fn with_audit_dir(mut self, dir: PathBuf) -> Self {
694        self.audit_dir = Some(dir);
695        self
696    }

698    /// Build Kubernetes target entries from the OpsConfig.
699    pub fn targets_from_config(config: &OpsConfig) -> Vec<KubeTarget> {
700        let mut all: Vec<&TargetConfig> = Vec::new();
701        all.extend(config.targets.as_deref().unwrap_or_default().iter());
702        for project in &config.projects {
703            for env in &project.environments {
704                all.extend(env.targets.iter());
705            }
706        }
707        all.into_iter()
708            .filter(|t| t.connection_type == ConnectionType::Kubernetes)
709            .map(|t| KubeTarget {
710                name: t.name.clone(),
711                kubeconfig: t.kubeconfig.clone(),
712                context: t.context.clone(),
713                default_namespace: t.namespace.clone(),
714                autonomy: t.autonomy,
715            })
716            .collect()
717    }
718
719    fn resolve<'a>(&'a self, name: &str) -> Option<&'a KubeTarget> {
720        self.config.targets.iter().find(|t| t.name == name)
721    }

723    fn audit(&self, target: &str, action: &str, detail: &str, duration_ms: u128, exit: i32) {
724        let _ = write_audit_entry(
725            target,
726            &format!("kube {action} {detail}"),
727            exit,
728            duration_ms,
729            self.audit_dir.as_ref(),
730        );
731    }

734/// Valid Kubernetes DNS-1123 name check — reject shell metacharacters even
735/// though we aren't shelling, to keep output predictable if we ever do.
736fn is_valid_k8s_name(s: &str) -> bool {
737    !s.is_empty()
738        && s.len() <= 253
739        && s.chars()
740            .all(|c| c.is_ascii_alphanumeric() || c == '-' || c == '.')
741}

743#[async_trait]
744impl Tool for KubeTool {
745    fn name(&self) -> &str {
746        "kubernetes"
747    }
748
749    fn description(&self) -> &str {
750        "Kubernetes cluster operations via the kube-rs client. Reads: \
751         cluster_info, list_namespaces, list_pods, list_deployments, \
752         list_services, list_nodes, logs (pod), events. Writes: \
753         restart_deployment, scale_deployment, delete_pod. Writes respect \
754         the target's autonomy level — DryRun rejects them. Every action \
755         is audit-logged."
756    }
757
758    fn parameters_schema(&self) -> Value {
759        json!({
760            "type": "object",
761            "properties": {
762                "target": {"type": "string", "description": "configured k8s target name"},
763                "action": {
764                    "type": "string",
765                    "enum": [
766                        "cluster_info", "list_namespaces", "list_pods",
767                        "list_deployments", "list_services", "list_nodes",
768                        "logs", "events",
769                        "restart_deployment", "scale_deployment", "delete_pod"
770                    ]
771                },
772                "namespace": {"type": "string"},
773                "name": {"type": "string", "description": "resource name"},
774                "lines": {"type": "integer", "default": 200, "description": "logs tail"},
775                "replicas": {"type": "integer", "description": "scale_deployment target"}
776            },
777            "required": ["target", "action"]
778        })
779    }
780
781    async fn execute(&self, args: Value) -> anyhow::Result<ToolResult> {
782        let target_name = match args.get("target").and_then(|v| v.as_str()) {
783            Some(s) => s,
784            None => return Ok(kube_err("missing 'target'")),
785        };
786        let target = match self.resolve(target_name) {
787            Some(t) => t,
788            None => return Ok(kube_err(format!("unknown target '{target_name}'"))),
789        };
790
791        let action = match args.get("action").and_then(|v| v.as_str()) {
792            Some(s) => s,
793            None => return Ok(kube_err("missing 'action'")),
794        };
795
796        let is_write = matches!(
797            action,
798            "restart_deployment" | "scale_deployment" | "delete_pod"
799        );
800        if is_write && target.autonomy == OpsClawAutonomy::DryRun {
801            let detail = args
802                .get("name")
803                .and_then(|v| v.as_str())
804                .unwrap_or("")
805                .to_string();
806            self.audit(
807                target_name,
808                &format!("[blocked dry-run] {action}"),
809                &detail,
810                0,
811                -1,
812            );
813            return Ok(kube_err(format!(
814                "dry-run mode: write action '{action}' rejected"
815            )));
816        }
817
818        let client = match self
819            .factory
820            .build(target.kubeconfig.as_deref(), target.context.as_deref())
821            .await
822        {
823            Ok(c) => c,
824            Err(e) => return Ok(kube_err(format!("kube client build failed: {e}"))),
825        };
826
827        let ns_default = target.default_namespace.as_deref();
828        let namespace = args
829            .get("namespace")
830            .and_then(|v| v.as_str())
831            .or(ns_default);
832
833        let start = std::time::Instant::now();
834        let result = self.dispatch(&client, action, namespace, &args).await;
835        let elapsed = start.elapsed().as_millis();
836        let exit = match &result {
837            Ok(r) if r.success => 0,
838            _ => 1,
839        };
840        let detail = args
841            .get("name")
842            .and_then(|v| v.as_str())
843            .unwrap_or("")
844            .to_string();
845        self.audit(target_name, action, &detail, elapsed, exit);
846        result
847    }
848}

850impl KubeTool {
851    async fn dispatch(
852        &self,
853        client: &KubeClient,
854        action: &str,
855        namespace: Option<&str>,
856        args: &Value,
857    ) -> anyhow::Result<ToolResult> {
858        match action {
859            "cluster_info" => match client.cluster_version().await {
860                Ok(v) => Ok(kube_ok(format!("version: {v}\n"))),
861                Err(e) => Ok(kube_err(format!("{e}"))),
862            },
863            "list_namespaces" => match client.list_namespaces().await {
864                Ok(nss) => {
865                    let mut out = String::new();
866                    writeln!(out, "count: {}", nss.len()).ok();
867                    for n in nss {
868                        writeln!(out, "  {n}").ok();
869                    }
870                    Ok(kube_ok(out))
871                }
872                Err(e) => Ok(kube_err(format!("{e}"))),
873            },
874            "list_pods" => match client.list_pods(namespace).await {
875                Ok(pods) => {
876                    let mut out = String::new();
877                    writeln!(out, "count: {}", pods.len()).ok();
878                    writeln!(out, "NS\tNAME\tREADY\tSTATUS\tRESTARTS\tNODE").ok();
879                    for p in pods {
880                        writeln!(
881                            out,
882                            "{}\t{}\t{}\t{}\t{}\t{}",
883                            p.namespace, p.name, p.ready, p.status, p.restarts, p.node
884                        )
885                        .ok();
886                    }
887                    Ok(kube_ok(out))
888                }
889                Err(e) => Ok(kube_err(format!("{e}"))),
890            },
891            "list_deployments" => match client.list_deployments(namespace).await {
892                Ok(deps) => {
893                    let mut out = String::new();
894                    writeln!(out, "count: {}", deps.len()).ok();
895                    writeln!(out, "NS\tNAME\tREADY\tUP-TO-DATE\tAVAILABLE").ok();
896                    for d in deps {
897                        writeln!(
898                            out,
899                            "{}\t{}\t{}\t{}\t{}",
900                            d.namespace, d.name, d.ready, d.up_to_date, d.available
901                        )
902                        .ok();
903                    }
904                    Ok(kube_ok(out))
905                }
906                Err(e) => Ok(kube_err(format!("{e}"))),
907            },
908            "list_services" => match client.list_services(namespace).await {
909                Ok(svcs) => {
910                    let mut out = String::new();
911                    writeln!(out, "count: {}", svcs.len()).ok();
912                    writeln!(out, "NS\tNAME\tTYPE\tCLUSTER-IP\tEXTERNAL-IP\tPORTS").ok();
913                    for s in svcs {
914                        writeln!(
915                            out,
916                            "{}\t{}\t{}\t{}\t{}\t{}",
917                            s.namespace, s.name, s.svc_type, s.cluster_ip, s.external_ip, s.ports
918                        )
919                        .ok();
920                    }
921                    Ok(kube_ok(out))
922                }
923                Err(e) => Ok(kube_err(format!("{e}"))),
924            },
925            "list_nodes" => match client.list_nodes().await {
926                Ok(nodes) => {
927                    let mut out = String::new();
928                    writeln!(out, "count: {}", nodes.len()).ok();
929                    writeln!(out, "NAME\tSTATUS\tROLES\tVERSION").ok();
930                    for n in nodes {
931                        writeln!(out, "{}\t{}\t{}\t{}", n.name, n.status, n.roles, n.version).ok();
932                    }
933                    Ok(kube_ok(out))
934                }
935                Err(e) => Ok(kube_err(format!("{e}"))),
936            },
937            "logs" => {
938                let name = match args.get("name").and_then(|v| v.as_str()) {
939                    Some(n) if is_valid_k8s_name(n) => n,
940                    Some(bad) => return Ok(kube_err(format!("invalid pod name '{bad}'"))),
941                    None => return Ok(kube_err("logs requires 'name'")),
942                };
943                let ns = match namespace {
944                    Some(n) => n,
945                    None => return Ok(kube_err("logs requires 'namespace' (or a default)")),
946                };
947                let lines = args.get("lines").and_then(|v| v.as_i64()).unwrap_or(200);
948                match client.get_pod_logs(ns, name, lines).await {
949                    Ok(logs) => Ok(kube_ok(logs)),
950                    Err(e) => Ok(kube_err(format!("{e}"))),
951                }
952            }
953            "events" => match client.get_events(namespace).await {
954                Ok(evts) => {
955                    let mut out = String::new();
956                    writeln!(out, "count: {}", evts.len()).ok();
957                    for e in evts {
958                        writeln!(
959                            out,
960                            "  {}  {}  {} — {}",
961                            e.timestamp, e.reason, e.involved_object, e.message
962                        )
963                        .ok();
964                    }
965                    Ok(kube_ok(out))
966                }
967                Err(e) => Ok(kube_err(format!("{e}"))),
968            },
969            "restart_deployment" => {
970                let name = match args.get("name").and_then(|v| v.as_str()) {
971                    Some(n) if is_valid_k8s_name(n) => n,
972                    Some(bad) => return Ok(kube_err(format!("invalid deployment name '{bad}'"))),
973                    None => return Ok(kube_err("restart_deployment requires 'name'")),
974                };
975                let ns = match namespace {
976                    Some(n) => n,
977                    None => return Ok(kube_err("restart_deployment requires 'namespace'")),
978                };
979                match client.restart_deployment(ns, name).await {
980                    Ok(()) => Ok(kube_ok(format!("restarted {ns}/{name}"))),
981                    Err(e) => Ok(kube_err(format!("{e}"))),
982                }
983            }
984            "scale_deployment" => {
985                let name = match args.get("name").and_then(|v| v.as_str()) {
986                    Some(n) if is_valid_k8s_name(n) => n,
987                    Some(bad) => return Ok(kube_err(format!("invalid deployment name '{bad}'"))),
988                    None => return Ok(kube_err("scale_deployment requires 'name'")),
989                };
990                let ns = match namespace {
991                    Some(n) => n,
992                    None => return Ok(kube_err("scale_deployment requires 'namespace'")),
993                };
994                let replicas = match args.get("replicas").and_then(|v| v.as_i64()) {
995                    Some(r) if (0..=10_000).contains(&r) => r as i32,
996                    _ => return Ok(kube_err("scale_deployment requires 'replicas' (0..=10000)")),
997                };
998                match client.scale_deployment(ns, name, replicas).await {
999                    Ok(()) => Ok(kube_ok(format!("scaled {ns}/{name} to {replicas}"))),
1000                    Err(e) => Ok(kube_err(format!("{e}"))),
1001                }
1002            }
1003            "delete_pod" => {
1004                let name = match args.get("name").and_then(|v| v.as_str()) {
1005                    Some(n) if is_valid_k8s_name(n) => n,
1006                    Some(bad) => return Ok(kube_err(format!("invalid pod name '{bad}'"))),
1007                    None => return Ok(kube_err("delete_pod requires 'name'")),
1008                };
1009                let ns = match namespace {
1010                    Some(n) => n,
1011                    None => return Ok(kube_err("delete_pod requires 'namespace'")),
1012                };
1013                match client.delete_pod(ns, name).await {
1014                    Ok(()) => Ok(kube_ok(format!("deleted pod {ns}/{name}"))),
1015                    Err(e) => Ok(kube_err(format!("{e}"))),
1016                }
1017            }
1018            other => Ok(kube_err(format!("unknown action '{other}'"))),
1019        }
1020    }
1021}
1022
1023fn kube_ok(mut s: String) -> ToolResult {
1024    if s.len() > KUBE_MAX_OUTPUT_BYTES {
1025        let mut cut = KUBE_MAX_OUTPUT_BYTES;
1026        while cut > 0 && !s.is_char_boundary(cut) {
1027            cut -= 1;
1028        }
1029        s.truncate(cut);
1030        s.push_str("\n... [truncated]");
1031    }
1032    ToolResult {
1033        success: true,
1034        output: s,
1035        error: None,
1036    }
1037}
1038
1039fn kube_err<S: Into<String>>(msg: S) -> ToolResult {
1040    ToolResult {
1041        success: false,
1042        output: String::new(),
1043        error: Some(msg.into()),
1044    }
1045}
