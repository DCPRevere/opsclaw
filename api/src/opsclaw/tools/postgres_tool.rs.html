1//! Postgres tool. Connects directly via tokio-postgres against configured
2//! instances. Read-only by default; writes (ANALYZE, VACUUM, REINDEX, and
3//! arbitrary EXEC) are gated by autonomy and audit-logged.
4//!
5//! Canned health actions: connections, locks, long_queries, replication,
6//! table_sizes, db_size. Plus query (SELECT) and exec (non-SELECT) for
7//! arbitrary statements.

45impl PostgresTool {
46    pub fn new(config: PostgresToolConfig) -> Self {
47        let instances = config
48            .instances
49            .iter()
50            .cloned()
51            .map(|i| (i.name.clone(), i))
52            .collect();
53        Self {
54            config,
55            instances,
56            audit_dir: None,
57        }
58    }
59
60    pub fn with_audit_dir(mut self, dir: PathBuf) -> Self {
61        self.audit_dir = Some(dir);
62        self
63    }
64
65    fn instance_names(&self) -> Vec<String> {
66        let mut v: Vec<String> = self.instances.keys().cloned().collect();
67        v.sort();
68        v
69    }
70
71    fn audit(&self, instance: &str, action: &str, detail: &str, duration_ms: u128, exit: i32) {
72        let _ = write_audit_entry(
73            &format!("postgres:{instance}"),
74            &format!("{action} {detail}"),
75            exit,
76            duration_ms,
77            self.audit_dir.as_ref(),
78        );
79    }
80}

82async fn connect(dsn: &str, timeout_ms: u64) -> anyhow::Result<Client> {
83    let (client, connection) = tokio::time::timeout(
84        Duration::from_millis(timeout_ms),
85        tokio_postgres::connect(dsn, NoTls),
86    )
87    .await
88    .map_err(|_| anyhow::anyhow!("connect timeout after {timeout_ms}ms"))??;
89
90    // Drive the connection future in the background. It ends when the
91    // client is dropped.
92    tokio::spawn(async move {
93        if let Err(e) = connection.await {
94            tracing::debug!("postgres connection closed: {e}");
95        }
96    });
97    Ok(client)
98}

100/// Safety guard for `query` action: refuse anything that doesn't start with
101/// SELECT, WITH, SHOW, or EXPLAIN. Strips comments/whitespace first.
102fn is_read_only_sql(sql: &str) -> bool {
103    let trimmed = strip_leading_comments(sql)
104        .trim_start()
105        .to_ascii_uppercase();
106    trimmed.starts_with("SELECT")
107        || trimmed.starts_with("WITH")
108        || trimmed.starts_with("SHOW")
109        || trimmed.starts_with("EXPLAIN")
110}
111
112fn strip_leading_comments(sql: &str) -> &str {
113    let mut s = sql.trim_start();
114    loop {
115        if let Some(rest) = s.strip_prefix("--") {
116            if let Some(nl) = rest.find('\n') {
117                s = &rest[nl + 1..].trim_start();
118            } else {
119                return "";
120            }
121        } else if let Some(rest) = s.strip_prefix("/*") {
122            if let Some(end) = rest.find("*/") {
123                s = &rest[end + 2..].trim_start();
124            } else {
125                return "";
126            }
127        } else {
128            return s;
129        }
130    }
131}

133/// Render a row as a compact TSV-ish line. Uses a small set of typed
134/// FromSql conversions; falls back to Debug for exotic types.
135fn render_value(row: &Row, idx: usize) -> String {
136    let col = &row.columns()[idx];
137    let ty = col.type_();
138    macro_rules! try_ty {
139        ($t:ty) => {
140            match row.try_get::<_, Option<$t>>(idx) {
141                Ok(Some(v)) => return format!("{v}"),
142                Ok(None) => return "NULL".into(),
143                Err(_) => {}
144            }
145        };
146    }
147    match *ty {
148        Type::BOOL => try_ty!(bool),
149        Type::INT2 => try_ty!(i16),
150        Type::INT4 => try_ty!(i32),
151        Type::INT8 => try_ty!(i64),
152        Type::FLOAT4 => try_ty!(f32),
153        Type::FLOAT8 => try_ty!(f64),
154        Type::TEXT | Type::VARCHAR | Type::NAME | Type::BPCHAR => try_ty!(String),
155        Type::TIMESTAMP | Type::TIMESTAMPTZ => try_ty!(chrono::NaiveDateTime),
156        Type::OID => try_ty!(u32),
157        _ => {}
158    }
159    // Fallback: try as string, else opaque
160    if let Ok(s) = row.try_get::<_, Option<String>>(idx) {
161        return s.unwrap_or_else(|| "NULL".into());
162    }
163    // Raw bytes — report type only.
164    format!("<{}>", ty.name())
165}

167fn render_rows(rows: &[Row], limit: i64) -> String {
168    let mut out = String::new();
169    writeln!(out, "rows: {}", rows.len()).ok();
170    if rows.is_empty() {
171        return out;
172    }
173    let cols = rows[0].columns();
174    let header = cols
175        .iter()
176        .map(|c| c.name().to_string())
177        .collect::<Vec<_>>()
178        .join("\t");
179    writeln!(out, "{header}").ok();
180    for (i, r) in rows.iter().enumerate() {
181        if i as i64 >= limit {
182            writeln!(out, "... [{} more rows]", rows.len() as i64 - limit).ok();
183            break;
184        }
185        let mut cells = Vec::with_capacity(cols.len());
186        for idx in 0..cols.len() {
187            cells.push(render_value(r, idx));
188        }
189        writeln!(out, "{}", cells.join("\t")).ok();
190    }
191    out
192}

194#[async_trait]
195impl Tool for PostgresTool {
196    fn name(&self) -> &str {
197        "postgres"
198    }
199
200    fn description(&self) -> &str {
201        "Postgres tool. Reads: connections, locks, long_queries, \
202         replication, table_sizes, db_size, query (SELECT/WITH/SHOW/\
203         EXPLAIN). Writes: exec (arbitrary SQL — gated by autonomy), \
204         analyze, vacuum, reindex. Writes respect the instance's autonomy \
205         — DryRun rejects them. Every statement is audit-logged. Direct \
206         driver connection via tokio-postgres; no shell involved."
207    }
208
209    fn parameters_schema(&self) -> Value {
210        json!({
211            "type": "object",
212            "properties": {
213                "instance": {"type": "string"},
214                "action": {
215                    "type": "string",
216                    "enum": [
217                        "connections", "locks", "long_queries", "replication",
218                        "table_sizes", "db_size", "query",
219                        "exec", "analyze", "vacuum", "reindex"
220                    ]
221                },
222                "sql": {"type": "string", "description": "query/exec"},
223                "limit": {"type": "integer", "default": 100},
224                "min_duration_ms": {"type": "integer", "default": 1000, "description": "long_queries"},
225                "table": {"type": "string", "description": "analyze/vacuum/reindex"},
226                "statement_timeout_ms": {"type": "integer", "default": 10000}
227            },
228            "required": ["instance", "action"]
229        })
230    }
231
232    async fn execute(&self, args: Value) -> anyhow::Result<ToolResult> {
233        let name = match args.get("instance").and_then(|v| v.as_str()) {
234            Some(s) => s,
235            None => return Ok(pg_err("missing 'instance'")),
236        };
237        let instance = match self.instances.get(name) {
238            Some(i) => i,
239            None => {
240                return Ok(pg_err(format!(
241                    "unknown instance '{name}'. Available: {}",
242                    self.instance_names().join(", ")
243                )));
244            }
245        };
246        let action = match args.get("action").and_then(|v| v.as_str()) {
247            Some(s) => s.to_string(),
248            None => return Ok(pg_err("missing 'action'")),
249        };
250
251        let is_write = matches!(action.as_str(), "exec" | "analyze" | "vacuum" | "reindex");
252        if is_write && instance.autonomy == OpsClawAutonomy::DryRun {
253            self.audit(name, &format!("[blocked dry-run] {action}"), "", 0, -1);
254            return Ok(pg_err(format!(
255                "dry-run mode: write action '{action}' rejected"
256            )));
257        }
258
259        let timeout_ms = args
260            .get("statement_timeout_ms")
261            .and_then(|v| v.as_u64())
262            .unwrap_or(DEFAULT_STATEMENT_TIMEOUT_MS);
263
264        let start = std::time::Instant::now();
265        let result = self.dispatch(instance, &action, &args, timeout_ms).await;
266        let elapsed = start.elapsed().as_millis();
267        let exit = match &result {
268            Ok(r) if r.success => 0,
269            _ => 1,
270        };
271        let detail = args
272            .get("sql")
273            .or_else(|| args.get("table"))
274            .and_then(|v| v.as_str())
275            .map(|s| s.chars().take(80).collect::<String>())
276            .unwrap_or_default();
277        self.audit(name, &action, &detail, elapsed, exit);
278        result
279    }
280}

282impl PostgresTool {
283    async fn dispatch(
284        &self,
285        instance: &PostgresInstance,
286        action: &str,
287        args: &Value,
288        timeout_ms: u64,
289    ) -> anyhow::Result<ToolResult> {
290        let client = match connect(&instance.dsn, timeout_ms).await {
291            Ok(c) => c,
292            Err(e) => return Ok(pg_err(format!("connect failed: {e}"))),
293        };
294        let _ = client
295            .simple_query(&format!("SET statement_timeout = {}", timeout_ms))
296            .await;
297
298        let limit = args
299            .get("limit")
300            .and_then(|v| v.as_i64())
301            .unwrap_or(DEFAULT_ROW_LIMIT)
302            .clamp(1, MAX_ROW_LIMIT);
303
304        match action {
305            "connections" => {
306                let sql = "SELECT state, count(*) AS n FROM pg_stat_activity GROUP BY state ORDER BY n DESC";
307                run_query(&client, sql, &[], limit).await
308            }
309            "locks" => {
310                let sql = "SELECT l.locktype, l.mode, l.granted, a.pid, a.usename, a.query \
311                    FROM pg_locks l JOIN pg_stat_activity a ON a.pid = l.pid \
312                    WHERE NOT l.granted OR l.mode LIKE '%Exclusive%' \
313                    ORDER BY l.granted, a.pid";
314                run_query(&client, sql, &[], limit).await
315            }
316            "long_queries" => {
317                let min_ms = args
318                    .get("min_duration_ms")
319                    .and_then(|v| v.as_i64())
320                    .unwrap_or(1000);
321                let sql = "SELECT pid, usename, state, now() - query_start AS duration, \
322                    substring(query, 1, 200) AS query \
323                    FROM pg_stat_activity \
324                    WHERE state <> 'idle' AND query_start IS NOT NULL \
325                      AND now() - query_start > make_interval(secs := $1::double precision / 1000) \
326                    ORDER BY duration DESC";
327                let min_ms_f64 = min_ms as f64;
328                run_query(&client, sql, &[&min_ms_f64], limit).await
329            }
330            "replication" => {
331                let sql = "SELECT application_name, state, sent_lsn, write_lsn, flush_lsn, \
332                    replay_lsn, write_lag, flush_lag, replay_lag \
333                    FROM pg_stat_replication";
334                run_query(&client, sql, &[], limit).await
335            }
336            "table_sizes" => {
337                let sql = "SELECT schemaname, relname, \
338                    pg_size_pretty(pg_total_relation_size(relid)) AS total_size, \
339                    pg_total_relation_size(relid) AS bytes \
340                    FROM pg_catalog.pg_statio_user_tables \
341                    ORDER BY pg_total_relation_size(relid) DESC";
342                run_query(&client, sql, &[], limit).await
343            }
344            "db_size" => {
345                let sql = "SELECT datname, pg_size_pretty(pg_database_size(datname)) AS size, \
346                    pg_database_size(datname) AS bytes FROM pg_database ORDER BY bytes DESC";
347                run_query(&client, sql, &[], limit).await
348            }
349            "query" => {
350                let sql = match args.get("sql").and_then(|v| v.as_str()) {
351                    Some(s) if !s.is_empty() => s,
352                    _ => return Ok(pg_err("query requires 'sql'")),
353                };
354                if !is_read_only_sql(sql) {
355                    return Ok(pg_err(
356                        "query action only allows SELECT/WITH/SHOW/EXPLAIN; use 'exec' for mutations",
357                    ));
358                }
359                run_query(&client, sql, &[], limit).await
360            }
361            "exec" => {
362                let sql = match args.get("sql").and_then(|v| v.as_str()) {
363                    Some(s) if !s.is_empty() => s,
364                    _ => return Ok(pg_err("exec requires 'sql'")),
365                };
366                match client.batch_execute(sql).await {
367                    Ok(()) => Ok(pg_ok("exec ok".into())),
368                    Err(e) => Ok(pg_err(format!("exec failed: {e}"))),
369                }
370            }
371            "analyze" => {
372                let sql = build_maintenance_sql("ANALYZE", args)?;
373                match client.batch_execute(&sql).await {
374                    Ok(()) => Ok(pg_ok(format!("ran: {sql}"))),
375                    Err(e) => Ok(pg_err(format!("{e}"))),
376                }
377            }
378            "vacuum" => {
379                let sql = build_maintenance_sql("VACUUM", args)?;
380                match client.batch_execute(&sql).await {
381                    Ok(()) => Ok(pg_ok(format!("ran: {sql}"))),
382                    Err(e) => Ok(pg_err(format!("{e}"))),
383                }
384            }
385            "reindex" => {
386                let sql = build_maintenance_sql("REINDEX TABLE", args)?;
387                match client.batch_execute(&sql).await {
388                    Ok(()) => Ok(pg_ok(format!("ran: {sql}"))),
389                    Err(e) => Ok(pg_err(format!("{e}"))),
390                }
391            }
392            other => Ok(pg_err(format!("unknown action '{other}'"))),
393        }
394    }
395}

397fn build_maintenance_sql(verb: &str, args: &Value) -> anyhow::Result<String> {
398    let table = args
399        .get("table")
400        .and_then(|v| v.as_str())
401        .ok_or_else(|| anyhow::anyhow!("{verb} requires 'table'"))?;
402    if !is_safe_table_ident(table) {
403        anyhow::bail!("invalid table identifier '{table}'");
404    }
405    Ok(format!("{verb} {table}"))
406}

408/// Accept `ident` or `schema.ident`. Letters, digits, `_`, one `.` allowed.
409fn is_safe_table_ident(s: &str) -> bool {
410    !s.is_empty()
411        && s.len() <= 128
412        && s.chars()
413            .all(|c| c.is_ascii_alphanumeric() || c == '_' || c == '.')
414        && s.matches('.').count() <= 1
415}

417async fn run_query(
418    client: &Client,
419    sql: &str,
420    params: &[&(dyn tokio_postgres::types::ToSql + Sync)],
421    limit: i64,
422) -> anyhow::Result<ToolResult> {
423    match client.query(sql, params).await {
424        Ok(rows) => Ok(pg_ok(render_rows(&rows, limit))),
425        Err(e) => Ok(pg_err(format!("{e}"))),
426    }
427}
428
429fn pg_ok(mut s: String) -> ToolResult {
430    if s.len() > MAX_OUTPUT_BYTES {
431        let mut cut = MAX_OUTPUT_BYTES;
432        while cut > 0 && !s.is_char_boundary(cut) {
433            cut -= 1;
434        }
435        s.truncate(cut);
436        s.push_str("\n... [truncated]");
437    }
438    ToolResult {
439        success: true,
440        output: s,
441        error: None,
442    }
443}
444
445fn pg_err<S: Into<String>>(msg: S) -> ToolResult {
446    ToolResult {
447        success: false,
448        output: String::new(),
449        error: Some(msg.into()),
450    }
451}
