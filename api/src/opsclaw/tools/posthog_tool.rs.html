1//! PostHog tool. Read-only v1.
2//!
3//! Surfaces product-analytics signal that closes the loop between an
4//! infra-side alert and what users are actually experiencing:
5//!
6//! - did the error event spike around the time the alert fired?
7//! - which feature flag was just rolled out?
8//! - what was *this specific user* doing when they reported the bug?
9//!
10//! Six actions, all GET/POST against the public PostHog API:
11//!
12//! - `query_events`         — count + sample for one event in a time window.
13//! - `recent_flag_changes`  — flags modified in the last N hours.
14//! - `flag_status`          — one flag: rollout %, recent metadata.
15//! - `events_for_user`      — last N events for a `distinct_id`.
16//! - `session_replay_url`   — most recent replay URL for a `distinct_id`.
17//! - `hogql`                — escape hatch for HogQL/ClickHouse queries.
18//!
19//! All actions are read-only; no write surface in v1.
20//! See `docs/tools/posthog.md` for the user-facing summary.
21

33const MAX_OUTPUT_BYTES: usize = 16 * 1024;
34/// Default cap on number of events surfaced per query. Without this, a
35/// careless agent can pull millions of rows on a wide HogQL query and spend
36/// real money on PostHog egress.
37const DEFAULT_LIMIT: u64 = 200;
38/// Hard ceiling — the agent can ask for up to this many; anything higher is
39/// silently clamped.
40const MAX_LIMIT: u64 = 5_000;
41/// Default time window when the agent doesn't specify one. SRE questions are
42/// almost always "since the alert fired" which is rarely longer than an hour.
43const DEFAULT_SINCE: &str = "-1h";

118    fn description(&self) -> &str {
119        "Query PostHog for user-facing signal that explains an SRE alert. \
120         Read-only. Reach for this when:\n\
121         - the alert is about a user flow ('checkout 5xx up', 'signup errors') \

123         - infra metrics moved with no obvious cause and you want to check whether \

125         - a customer reported a bug and you need their session replay or recent \

127         - you're correlating a deploy with a behaviour change.\n\
128         Actions: query_events, recent_flag_changes, flag_status, \

152                "limit":        {"type": "integer", "description": "max rows; default 200, clamped to 5000"},

156        })

158
159    async fn execute(&self, args: Value) -> anyhow::Result<ToolResult> {
160        let action = match args.get("action").and_then(|v| v.as_str()) {
161            Some(s) => s.to_string(),
162            None => return Ok(err("missing 'action'")),
163        };
164        let start = std::time::Instant::now();
165        let result = self.dispatch(&action, &args).await;
166        let elapsed = start.elapsed().as_millis();
167        let exit = match &result {
168            Ok(r) if r.success => 0,
169            _ => 1,
170        };
171        let detail = action_detail(&action, &args);
172        self.audit(&action, &detail, elapsed, exit);
173        result
174    }
175}

177impl PostHogTool {
178    async fn dispatch(&self, action: &str, args: &Value) -> anyhow::Result<ToolResult> {
179        match action {
180            "query_events" => self.query_events(args).await,
181            "recent_flag_changes" => self.recent_flag_changes(args).await,
182            "flag_status" => self.flag_status(args).await,
183            "events_for_user" => self.events_for_user(args).await,
184            "session_replay_url" => self.session_replay_url(args).await,
185            "hogql" => self.hogql_query(args).await,
186            other => Ok(err(format!("unknown action '{other}'"))),
187        }
188    }

190    /// Count + sample for one event over a time window. Property filters
191    /// are AND-ed with equality.
192    async fn query_events(&self, args: &Value) -> anyhow::Result<ToolResult> {
193        let event = match args.get("event_name").and_then(|v| v.as_str()) {
194            Some(s) if !s.is_empty() => s.to_string(),
195            _ => return Ok(err("missing 'event_name'")),
196        };
197        let since = since_or_default(args);
198        let until = until_or_default(args);
199        let limit = clamp_limit(args);
200
201        // Build a HogQL query: count + small sample. Property filters land in
202        // the WHERE via parameterised equality; we don't accept raw SQL there
203        // to keep the agent's surface narrow.
204        let mut where_clauses: Vec<String> = vec![
205            format!("event = '{}'", escape_sql(&event)),
206            format!("timestamp >= toDateTime('{since}')"),
207            format!("timestamp <= toDateTime('{until}')"),
208        ];
209        if let Some(filters) = args.get("filters").and_then(|v| v.as_object()) {
210            for (k, v) in filters {
211                if let Some(val) = v.as_str() {
212                    where_clauses.push(format!(
213                        "properties['{}'] = '{}'",
214                        escape_sql(k),
215                        escape_sql(val)
216                    ));
217                }
218            }
219        }
220        let where_sql = where_clauses.join(" AND ");
221
222        // Two queries: total count, plus a small sample so the agent has
223        // something concrete to reason about.
224        let count_sql = format!("SELECT count() FROM events WHERE {where_sql}");
225        let sample_sql = format!(
226            "SELECT timestamp, distinct_id, properties FROM events \
227             WHERE {where_sql} ORDER BY timestamp DESC LIMIT {limit}"
228        );
229
230        let count = self.run_hogql(&count_sql).await?;
231        let sample = self.run_hogql(&sample_sql).await?;
232
233        let total = count
234            .get("results")
235            .and_then(|r| r.as_array())
236            .and_then(|rows| rows.first())
237            .and_then(|row| row.as_array())
238            .and_then(|cols| cols.first())
239            .and_then(|c| c.as_u64())
240            .unwrap_or(0);
241
242        let mut out = String::new();
243        writeln!(out, "event: {event}").ok();
244        writeln!(out, "window: {since} → {until}").ok();
245        writeln!(out, "total: {total}").ok();
246        writeln!(out, "sample (up to {limit}):").ok();
247        if let Some(rows) = sample.get("results").and_then(|r| r.as_array()) {
248            for row in rows.iter().take(limit as usize) {
249                if let Some(cols) = row.as_array() {
250                    let ts = cols.first().and_then(|v| v.as_str()).unwrap_or("?");
251                    let did = cols.get(1).and_then(|v| v.as_str()).unwrap_or("?");
252                    let props = cols
253                        .get(2)
254                        .map(|v| serde_json::to_string(v).unwrap_or_default())
255                        .unwrap_or_default();
256                    let props_short = &props[..props.len().min(200)];
257                    writeln!(out, "  {ts} {did} {props_short}").ok();
258                }
259            }
260        }
261        Ok(ok_res(out))
262    }

264    /// Flags modified in the last N hours, optionally limited to active.
265    async fn recent_flag_changes(&self, args: &Value) -> anyhow::Result<ToolResult> {
266        let hours = args.get("hours").and_then(|v| v.as_u64()).unwrap_or(24);
267        let active_only = args
268            .get("active_only")
269            .and_then(|v| v.as_bool())
270            .unwrap_or(true);
271
272        // PostHog's feature_flags endpoint returns all flags; we filter
273        // client-side by `updated_at` against the cutoff.
274        let resp = self
275            .req(reqwest::Method::GET, "feature_flags")
276            .send()
277            .await?;
278        let (ok, status, body) = consume(resp).await;
279        if !ok {
280            return Ok(err(format!("{status}: {}", snippet(&body))));
281        }
282        let v: Value = serde_json::from_str(&body).unwrap_or(Value::Null);
283        let flags = match v.get("results").and_then(|r| r.as_array()) {
284            Some(arr) => arr,
285            None => return Ok(ok_res("no flags returned\n".into())),
286        };
287
288        let cutoff = chrono::Utc::now() - chrono::Duration::hours(hours as i64);
289        let cutoff_str = cutoff.to_rfc3339();
290
291        let mut out = String::new();
292        writeln!(out, "flags modified since {cutoff_str}:").ok();
293        let mut shown = 0usize;
294        for f in flags {
295            let updated = f.get("updated_at").and_then(|v| v.as_str()).unwrap_or("");
296            if updated.is_empty() || updated < cutoff_str.as_str() {
297                continue;
298            }
299            let active = f.get("active").and_then(|v| v.as_bool()).unwrap_or(false);
300            if active_only && !active {
301                continue;
302            }
303            let key = f.get("key").and_then(|v| v.as_str()).unwrap_or("");
304            let rollout = f
305                .get("rollout_percentage")
306                .and_then(|v| v.as_u64())
307                .map(|p| format!("{p}%"))
308                .unwrap_or_else(|| "—".into());
309            writeln!(
310                out,
311                "  {key} active={active} rollout={rollout} updated={updated}"
312            )
313            .ok();
314            shown += 1;
315        }
316        if shown == 0 {
317            writeln!(out, "  (none)").ok();
318        }
319        Ok(ok_res(out))
320    }

322    /// One flag's full metadata: rollout %, filters, active state.
323    async fn flag_status(&self, args: &Value) -> anyhow::Result<ToolResult> {
324        let key = match args.get("flag_key").and_then(|v| v.as_str()) {
325            Some(s) if !s.is_empty() => s.to_string(),
326            _ => return Ok(err("missing 'flag_key'")),
327        };
328        // PostHog's feature_flags endpoint supports ?key=, but only some
329        // versions; safer to GET all and filter.
330        let resp = self
331            .req(reqwest::Method::GET, "feature_flags")
332            .send()
333            .await?;
334        let (ok, status, body) = consume(resp).await;
335        if !ok {
336            return Ok(err(format!("{status}: {}", snippet(&body))));
337        }
338        let v: Value = serde_json::from_str(&body).unwrap_or(Value::Null);
339        let flag = v.get("results").and_then(|r| r.as_array()).and_then(|arr| {
340            arr.iter()
341                .find(|f| f.get("key").and_then(|k| k.as_str()) == Some(key.as_str()))
342        });
343        let Some(flag) = flag else {
344            return Ok(err(format!("flag '{key}' not found")));
345        };
346
347        let mut out = String::new();
348        writeln!(out, "flag: {key}").ok();
349        writeln!(
350            out,
351            "  active: {}",
352            flag.get("active")
353                .and_then(|v| v.as_bool())
354                .unwrap_or(false)
355        )
356        .ok();
357        if let Some(p) = flag.get("rollout_percentage").and_then(|v| v.as_u64()) {
358            writeln!(out, "  rollout: {p}%").ok();
359        }
360        if let Some(updated) = flag.get("updated_at").and_then(|v| v.as_str()) {
361            writeln!(out, "  updated_at: {updated}").ok();
362        }
363        if let Some(created) = flag.get("created_at").and_then(|v| v.as_str()) {
364            writeln!(out, "  created_at: {created}").ok();
365        }
366        if let Some(name) = flag.get("name").and_then(|v| v.as_str()) {
367            if !name.is_empty() {
368                writeln!(out, "  name: {name}").ok();
369            }
370        }
371        if let Some(filters) = flag.get("filters") {
372            let pretty = serde_json::to_string_pretty(filters).unwrap_or_default();
373            writeln!(out, "  filters:\n{}", indent(&pretty, 4)).ok();
374        }
375        Ok(ok_res(out))
376    }

378    /// Last N events for a single `distinct_id`. The classic "user reported
379    /// a bug — what were they doing" query.
380    async fn events_for_user(&self, args: &Value) -> anyhow::Result<ToolResult> {
381        let did = match args.get("distinct_id").and_then(|v| v.as_str()) {
382            Some(s) if !s.is_empty() => s.to_string(),
383            _ => return Ok(err("missing 'distinct_id'")),
384        };
385        let limit = clamp_limit(args);
386        let since = since_or_default(args);
387
388        let sql = format!(
389            "SELECT timestamp, event, properties FROM events \
390             WHERE distinct_id = '{}' AND timestamp >= toDateTime('{since}') \
391             ORDER BY timestamp DESC LIMIT {limit}",
392            escape_sql(&did)
393        );
394        let result = self.run_hogql(&sql).await?;
395        let mut out = String::new();
396        writeln!(out, "distinct_id: {did}").ok();
397        writeln!(out, "since: {since}").ok();
398        if let Some(rows) = result.get("results").and_then(|r| r.as_array()) {
399            writeln!(out, "events ({}):", rows.len()).ok();
400            for row in rows {
401                if let Some(cols) = row.as_array() {
402                    let ts = cols.first().and_then(|v| v.as_str()).unwrap_or("?");
403                    let event = cols.get(1).and_then(|v| v.as_str()).unwrap_or("?");
404                    let props = cols
405                        .get(2)
406                        .map(|v| serde_json::to_string(v).unwrap_or_default())
407                        .unwrap_or_default();
408                    let props_short = &props[..props.len().min(150)];
409                    writeln!(out, "  {ts} {event} {props_short}").ok();
410                }
411            }
412        }
413        Ok(ok_res(out))
414    }

416    /// Most recent session replay for a `distinct_id`. Returns the URL the
417    /// human can paste into a browser; replays themselves are too big to
418    /// inline.
419    async fn session_replay_url(&self, args: &Value) -> anyhow::Result<ToolResult> {
420        let did = match args.get("distinct_id").and_then(|v| v.as_str()) {
421            Some(s) if !s.is_empty() => s.to_string(),
422            _ => return Ok(err("missing 'distinct_id'")),
423        };
424        let resp = self
425            .req(reqwest::Method::GET, "session_recordings")
426            .query(&[("distinct_id", did.as_str()), ("limit", "1")])
427            .send()
428            .await?;
429        let (ok, status, body) = consume(resp).await;
430        if !ok {
431            return Ok(err(format!("{status}: {}", snippet(&body))));
432        }
433        let v: Value = serde_json::from_str(&body).unwrap_or(Value::Null);
434        let recording = v
435            .get("results")
436            .and_then(|r| r.as_array())
437            .and_then(|arr| arr.first());
438        let Some(rec) = recording else {
439            return Ok(ok_res(format!("no replays found for distinct_id={did}\n")));
440        };
441
442        let id = rec.get("id").and_then(|v| v.as_str()).unwrap_or("?");
443        let start = rec
444            .get("start_time")
445            .and_then(|v| v.as_str())
446            .unwrap_or("?");
447        let duration = rec
448            .get("recording_duration")
449            .and_then(|v| v.as_u64())
450            .unwrap_or(0);
451        let url = format!(
452            "{}/project/{}/replay/{}",
453            self.config.host.trim_end_matches('/'),
454            self.config.project_id,
455            id
456        );
457        let mut out = String::new();
458        writeln!(out, "distinct_id: {did}").ok();
459        writeln!(out, "replay: {url}").ok();
460        writeln!(out, "  start: {start}").ok();
461        writeln!(out, "  duration_s: {duration}").ok();
462        Ok(ok_res(out))
463    }

465    /// Run a raw HogQL query. Power-user escape hatch; the agent should
466    /// reach for the structured actions first.
467    ///
468    /// Guarded: only SELECT/WITH queries are accepted. PostHog's HogQL is
469    /// read-only by design today, but the guard keeps us safe against future
470    /// surface expansion (e.g. INSERT INTO events) and against accidental
471    /// foot-guns from the agent.
472    async fn hogql_query(&self, args: &Value) -> anyhow::Result<ToolResult> {
473        let query = match args.get("query").and_then(|v| v.as_str()) {
474            Some(s) if !s.is_empty() => s.to_string(),
475            _ => return Ok(err("missing 'query'")),
476        };
477        if let Err(reason) = ensure_select_only(&query) {
478            return Ok(err(format!(
479                "hogql query rejected: {reason}. Only SELECT/WITH queries are allowed."
480            )));
481        }
482        let result = self.run_hogql(&query).await?;
483        let pretty = serde_json::to_string_pretty(&result).unwrap_or_default();
484        Ok(ok_res(pretty))
485    }
486
487    /// Internal: POST to /query with a HogQL body. Single API entry point
488    /// for everything that needs ClickHouse-flavoured access.
489    async fn run_hogql(&self, query: &str) -> anyhow::Result<Value> {
490        let body = json!({
491            "query": {
492                "kind": "HogQLQuery",
493                "query": query,
494            }
495        });
496        let resp = self
497            .req(reqwest::Method::POST, "query")
498            .json(&body)
499            .send()
500            .await?;
501        let (ok, status, raw) = consume(resp).await;
502        if !ok {
503            anyhow::bail!("hogql {status}: {}", snippet(&raw));
504        }
505        Ok(serde_json::from_str(&raw).unwrap_or(Value::Null))
506    }

558/// Minimal SQL-string escape for HogQL string literals. Doubles single
559/// quotes; rejects nothing — HogQL will surface its own parser errors for
560/// anything truly malformed. Works because we only build literal-equality
561/// `=` clauses; we don't accept raw SQL fragments from the agent for the
562/// structured actions.
563fn escape_sql(s: &str) -> String {
564    s.replace('\'', "''")
565}
566
567/// Enforce that a HogQL query is read-only. Strips comments, lowercases,
568/// then checks the first non-whitespace token is `select` or `with`, and
569/// that no DML/DDL keyword appears as a standalone token elsewhere.
570///
571/// PostHog's HogQL is read-only by design today, but this guard means we
572/// don't have to revisit it if that ever changes. False positives are
573/// possible on contrived strings (e.g. a column literally named `delete`)
574/// but the structured actions cover the common cases — `hogql` is the
575/// power-user escape hatch and a strict guard is appropriate.
576fn ensure_select_only(query: &str) -> Result<(), &'static str> {
577    let lower = strip_sql_comments(query).to_ascii_lowercase();
578    let trimmed = lower.trim();
579    if trimmed.is_empty() {
580        return Err("empty query");
581    }
582    let first_token = trimmed
583        .split(|c: char| c.is_whitespace() || c == '(')
584        .find(|s| !s.is_empty())
585        .unwrap_or("");
586    if !matches!(first_token, "select" | "with") {
587        return Err("query must start with SELECT or WITH");
588    }
589    const FORBIDDEN: &[&str] = &[
590        "insert", "update", "delete", "drop", "alter", "truncate", "create", "replace", "grant",
591        "revoke", "attach", "detach", "rename", "optimize",
592    ];
593    for word in FORBIDDEN {
594        // Word-boundary scan: needle must be surrounded by non-word chars.
595        let needle = format!(" {word} ");
596        let padded = format!(" {trimmed} ");
597        if padded.contains(&needle) {
598            return Err("contains forbidden DML/DDL keyword");
599        }
600    }
601    Ok(())
602}
603
604fn strip_sql_comments(s: &str) -> String {
605    // Drop everything from `--` to end-of-line and `/* ... */` blocks. Naive
606    // but adequate for a guard — the worst case is we leave a comment in,
607    // which only widens what we accept (still read-only).
608    let mut out = String::with_capacity(s.len());
609    let mut chars = s.chars().peekable();
610    while let Some(c) = chars.next() {
611        match c {
612            '-' if chars.peek() == Some(&'-') => {
613                while let Some(&n) = chars.peek() {
614                    chars.next();
615                    if n == '\n' {
616                        out.push('\n');
617                        break;
618                    }
619                }
620            }
621            '/' if chars.peek() == Some(&'*') => {
622                chars.next();
623                while let Some(n) = chars.next() {
624                    if n == '*' && chars.peek() == Some(&'/') {
625                        chars.next();
626                        break;
627                    }
628                }
629            }
630            _ => out.push(c),
631        }
632    }
633    out
634}
635
636fn indent(s: &str, n: usize) -> String {
637    let pad = " ".repeat(n);
638    s.lines()
639        .map(|l| format!("{pad}{l}"))
640        .collect::<Vec<_>>()
641        .join("\n")
642}

662fn ok_res(mut s: String) -> ToolResult {
663    if s.len() > MAX_OUTPUT_BYTES {
664        let mut cut = MAX_OUTPUT_BYTES;
665        while cut > 0 && !s.is_char_boundary(cut) {
666            cut -= 1;
667        }
668        s.truncate(cut);
669        s.push_str("\n... [truncated]");
670    }
671    ToolResult {
672        success: true,
673        output: s,
674        error: None,
675    }
676}
